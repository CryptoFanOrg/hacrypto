48 static void compute_kbltz(void) {
49  int u, i;
50  bn_t a, b, c;
51  ctx_t *ctx = core_get();
52 
53  bn_null(a);
54  bn_null(b);
55  bn_null(c);
56 
57  TRY {
58  bn_new(a);
59  bn_new(b);
60  bn_new(c);
61 
62  if (ctx->eb_opt_a == OPT_ZERO) {
63  u = -1;
64  } else {
65  u = 1;
66  }
67 
68  bn_set_dig(a, 2);
69  bn_set_dig(b, 1);
70  if (u == -1) {
71  bn_neg(b, b);
72  }
73  for (i = 2; i <= FB_BITS; i++) {
74  bn_copy(c, b);
75  if (u == -1) {
76  bn_neg(b, b);
77  }
78  bn_dbl(a, a);
79  bn_sub(b, b, a);
80  bn_copy(a, c);
81  }
82  bn_copy(&(ctx->eb_vm), b);
83 
84  bn_zero(a);
85  bn_set_dig(b, 1);
86  for (i = 2; i <= FB_BITS; i++) {
87  bn_copy(c, b);
88  if (u == -1) {
89  bn_neg(b, b);
90  }
91  bn_dbl(a, a);
92  bn_sub(b, b, a);
93  bn_add_dig(b, b, 1);
94  bn_copy(a, c);
95  }
96  bn_copy(&(ctx->eb_s0), b);
97 
98  bn_zero(a);
99  bn_zero(b);
100  for (i = 2; i <= FB_BITS; i++) {
101  bn_copy(c, b);
102  if (u == -1) {
103  bn_neg(b, b);
104  }
105  bn_dbl(a, a);
106  bn_sub(b, b, a);
107  bn_sub_dig(b, b, 1);
108  bn_copy(a, c);
109  }
110  bn_copy(&(ctx->eb_s1), b);
111 
112  }
113  CATCH_ANY {
114  THROW(ERR_CAUGHT);
115  }
116  FINALLY {
117  bn_free(a);
118  bn_free(b);
119  bn_free(c);
120  }
121 }

131 static void detect_opt(int *opt, fb_t a) {
132  if (fb_is_zero(a)) {
133  *opt = OPT_ZERO;
134  } else {
135  if (fb_cmp_dig(a, 1) == CMP_EQ) {
136  *opt = OPT_ONE;
137  } else {
138  if (fb_bits(a) <= FB_DIGIT) {
139  *opt = OPT_DIGIT;
140  } else {
141  *opt = OPT_NONE;
142  }
143  }
144  }
