355 void ep_param_set(int param) {

357  char str[2 * FP_BYTES + 2];
358  fp_t a, b, beta;

361

364  fp_null(beta);
365  bn_null(lamb);
366  ep_null(g);
367  bn_null(r);
368  bn_null(h);
369 
370  TRY {

373  fp_new(beta);
374  bn_new(lamb);
375  ep_new(g);
