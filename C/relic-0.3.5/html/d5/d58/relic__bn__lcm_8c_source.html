38 void bn_lcm(bn_t c, const bn_t a, const bn_t b) {
39  bn_t u, v;
40 
41  bn_null(u);
42  bn_null(v);
43 
44  TRY {
45  bn_new(u);
46  bn_new(v);
47 
48  bn_gcd(u, a, b);
49 
50  if (bn_cmp_abs(a, b) == CMP_LT) {
51  bn_div(v, a, u);
52  bn_mul(c, b, v);
53  } else {
54  bn_div(v, b, u);
55  bn_mul(c, a, v);
56  }
57  c->sign = BN_POS;
58  }
59  CATCH_ANY {
60  THROW(ERR_CAUGHT);
61  }
62  FINALLY {
63  bn_free(u);
64  bn_free(v);
65  }
66 }
