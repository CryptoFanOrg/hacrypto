189  fp2_t t0, t1, t2, t3, t4;
190  int one = 1, zero = 0;
191 
192  fp2_null(t0);
193  fp2_null(t1);
194  fp2_null(t2);
195  fp2_null(t3);
196  fp2_null(t4);
197 
198  TRY {
199  fp2_new(t0);
200  fp2_new(t1);
201  fp2_new(t2);
202  fp2_new(t3);
203  fp2_new(t4);
204 
205  /* B = t0 = x1 - x2 * z1. */
206  fp2_mul(t0, r->z, q->x);
207  fp2_sub(t0, r->x, t0);
208  /* A = t1 = y1 - y2 * z1. */
209  fp2_mul(t1, r->z, q->y);
210  fp2_sub(t1, r->y, t1);
211 
212  /* D = B^2. */
213  fp2_sqr(t2, t0);
214  /* G = x1 * D. */
215  fp2_mul(r->x, r->x, t2);
216  /* E = B^3. */
217  fp2_mul(t2, t2, t0);
218  /* C = A^2. */
219  fp2_sqr(t3, t1);
220  /* F = E + z1 * C. */
221  fp2_mul(t3, t3, r->z);
222  fp2_add(t3, t2, t3);
223 
224  if (ep2_curve_is_twist() == EP_MTYPE) {
225  one ^= 1;
226  zero ^= 1;
227  }
228 
229  /* l10 = - (A * xp). */
230  fp_mul(l[one][zero][0], t1[0], p->x);
231  fp_mul(l[one][zero][1], t1[1], p->x);
232  fp2_neg(l[one][zero], l[one][zero]);
233 
234  /* t4 = B * x2. */
235  fp2_mul(t4, q->x, t1);
236 
237  /* H = E + F - 2 * G. */
238  fp2_sub(t3, t3, r->x);
239  fp2_sub(t3, t3, r->x);
240  /* y3 = A * (G - H) - y1 * E. */
241  fp2_sub(r->x, r->x, t3);
242  fp2_mul(t1, t1, r->x);
243  fp2_mul(r->y, t2, r->y);
244  fp2_sub(r->y, t1, r->y);
245  /* x3 = B * H. */
246  fp2_mul(r->x, t0, t3);
247  /* z3 = z1 * E. */
248  fp2_mul(r->z, r->z, t2);
249 
250  /* l11 = J = B * x2 - A * y2. */
251  fp2_mul(t2, q->y, t0);
252  fp2_sub(l[one][one], t4, t2);
253 
254  /* l00 = B * yp. */
255  fp_mul(l[zero][zero][0], t0[0], p->y);
256  fp_mul(l[zero][zero][1], t0[1], p->y);
257 
258  r->norm = 0;
259  }
260  CATCH_ANY {
261  THROW(ERR_CAUGHT);
262  } FINALLY {
263  fp2_free(t0);
264  fp2_free(t1);
265  fp2_free(t2);
266  fp2_free(t3);
267  fp2_free(t4);
268  }
