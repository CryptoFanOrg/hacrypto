50 void pp_norm_imp(ep2_t r, ep2_t p) {
51  fp2_inv(r->z, p->z);
52  fp2_mul(r->x, p->x, r->z);
53  fp2_mul(r->y, p->y, r->z);
54  fp_zero(r->z[0]);
