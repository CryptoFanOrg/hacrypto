54 static void fb_mul_basic_imp(dig_t *c, const dig_t *a, const dig_t *b, int size) {
55  int i;
56  dv_t s;
57 
58  dv_null(s);
59 
60  TRY {
61  /* We need a temporary variable so that c can be a or b. */
62  dv_new(s);
63  dv_zero(s, 2 * FB_DIGS);
64 
65  dv_copy(s, b, size);
66  dv_zero(c, 2 * size);
67 
68  if (a[0] & 1) {
69  dv_copy(c, b, size);
70  }
71  for (i = 1; i <= (FB_DIGIT * size) - 1; i++) {
72  fb_lsh1_low(s, s);
73  fb_rdc(s, s);
74  if (fb_test_bit(a, i)) {
75  fb_add(c, c, s);
76  }
77  }
78  }
79  CATCH_ANY {
80  THROW(ERR_CAUGHT);
81  }
82  FINALLY {
83  dv_free(s);
84  }
85 }

97 static void fb_mul_lcomb_imp(dig_t *c, const dig_t *a, const dig_t *b, int size) {
98  dv_zero(c, 2 * size);
99 
100  for (int i = FB_DIGIT - 1; i >= 0; i--) {
101  for (int j = 0; j < size; j++) {
102  if (a[j] & ((dig_t)1 << i)) {
103  fb_addd_low(c + j, c + j, b, size);
104  }
105  }
106  if (i != 0) {
107  bn_lsh1_low(c, c, 2 * size);
108  }
109  }

166 static void fb_mul_karat_imp(dv_t c, const fb_t a, const fb_t b, int size,
167  int level) {
168  int i, h, h1;
169  dv_t a1, b1, ab;
170  dig_t *a0b0, *a1b1;
171 
172  dv_null(a1);
173  dv_null(b1);
174  dv_null(ab);
175 
176  /* Compute half the digits of a or b. */
177  h = size >> 1;
178  h1 = size - h;
179 
180  TRY {
181  /* Allocate the temp variables. */
182  dv_new(a1);
183  dv_new(b1);
184  dv_new(ab);
185  a0b0 = ab;
186  a1b1 = ab + 2 * h;
187 
188  /* a0b0 = a0 * b0 and a1b1 = a1 * b1 */
189  if (level <= 1) {
190 #if FB_MUL == BASIC
191  fb_mul_basic_imp(a0b0, a, b, h);
192  fb_mul_basic_imp(a1b1, a + h, b + h, h1);
193 #elif FB_MUL == LCOMB
194  fb_mul_lcomb_imp(a0b0, a, b, h);
195  fb_mul_lcomb_imp(a1b1, a + h, b + h, h1);
196 #elif FB_MUL == RCOMB
197  fb_mul_rcomb_imp(a0b0, a, b, h);
198  fb_mul_rcomb_imp(a1b1, a + h, b + h, h1);
199 #elif FB_MUL == INTEG || FB_MUL == LODAH
200  fb_muld_low(a0b0, a, b, h);
201  fb_muld_low(a1b1, a + h, b + h, h1);
202 #endif
203  } else {
204  fb_mul_karat_imp(a0b0, a, b, h, level - 1);
205  fb_mul_karat_imp(a1b1, a + h, b + h, h1, level - 1);
206  }
207 
208  for (i = 0; i < 2 * size; i++) {
209  c[i] = ab[i];
210  }
211 
212  /* c = c - (a0*b0 << h digits) */
213  fb_addd_low(c + h, c + h, a0b0, 2 * h);
214 
215  /* c = c - (a1*b1 << h digits) */
216  fb_addd_low(c + h, c + h, a1b1, 2 * h1);
217 
218  /* a1 = (a1 + a0) */
219  fb_addd_low(a1, a, a + h, h);
220 
221  /* b1 = (b1 + b0) */
222  fb_addd_low(b1, b, b + h, h);
223 
224  if (h1 > h) {
225  a1[h1 - 1] = a[h + h1 - 1];
226  b1[h1 - 1] = b[h + h1 - 1];
227  }
228 
229  if (level <= 1) {
230  /* a1b1 = (a1 + a0)*(b1 + b0) */
231 #if FB_MUL == BASIC
232  fb_mul_basic_imp(a1b1, a1, b1, h1);
233 #elif FB_MUL == LCOMB
234  fb_mul_lcomb_imp(a1b1, a1, b1, h1);
235 #elif FB_MUL == RCOMB
236  fb_mul_rcomb_imp(a1b1, a1, b1, h1);
237 #elif FB_MUL == INTEG || FB_MUL == LODAH
238  fb_muld_low(a1b1, a1, b1, h1);
239 #endif
240  } else {
241  fb_mul_karat_imp(a1b1, a1, b1, h1, level - 1);
242  }
243 
244  /* c = c + [(a1 + a0)*(b1 + b0) << digits] */
245  fb_addd_low(c + h, c + h, a1b1, 2 * h1);
246  }
247  CATCH_ANY {
248  THROW(ERR_CAUGHT);
249  }
250  FINALLY {
251  dv_free(a1);
252  dv_free(b1);
253  dv_free(ab);
254  }
255 }

265 void fb_mul_basic(fb_t c, const fb_t a, const fb_t b) {
266  int i;
267  dv_t s;
268  fb_t t;
269 
270  dv_null(s);
271  fb_null(t);
272 
273  TRY {
274  /* We need a temporary variable so that c can be a or b. */
275  fb_new(t);
276  dv_new(s);
277  fb_zero(t);
278  dv_zero(s + FB_DIGS, FB_DIGS);
279  fb_copy(s, b);
280 
281  if (a[0] & 1) {
282  fb_copy(t, b);
283  }
284  for (i = 1; i < FB_BITS; i++) {
285  /* We are already shifting a temporary value, so this is more efficient
286  * than calling fb_lsh(). */
287  s[FB_DIGS] = fb_lsh1_low(s, s);
288  fb_rdc(s, s);
289  if (fb_test_bit(a, i)) {
290  fb_add(t, t, s);
291  }
292  }
293 
294  if (fb_bits(t) > FB_BITS) {
295  fb_poly_add(c, t);
296  } else {
297  fb_copy(c, t);
298  }
299  }
300  CATCH_ANY {
301  THROW(ERR_CAUGHT);
302  }
303  FINALLY {
304  fb_free(t);
305  fb_free(s);
306  }
307 }

321 void fb_mul_lcomb(fb_t c, const fb_t a, const fb_t b) {
322  dv_t t;
323  dig_t carry;
324 
325  dv_null(t);
326 
327  TRY {
328  dv_new(t);
329  dv_zero(t, 2 * FB_DIGS);
330 
331  for (int i = FB_DIGIT - 1; i >= 0; i--) {
332  for (int j = 0; j < FB_DIGS; j++) {
333  if (a[j] & ((dig_t)1 << i)) {
334  /* This cannot use fb_addn_low() because there is no
335  * guarantee that operands will be aligned. */
336  fb_addd_low(t + j, t + j, b, FB_DIGS);
337  }
338  }
339  if (i != 0) {
340  carry = fb_lsh1_low(t, t);
341  fb_lsh1_low(t + FB_DIGS, t + FB_DIGS);
342  t[FB_DIGS] |= carry;
343  }
344  }
345 
346  fb_rdc(c, t);
347  }
348  CATCH_ANY {
349  THROW(ERR_CAUGHT);
350  }
351  FINALLY {
352  dv_free(t);
353  }
354 }

363 
364  dv_null(t);

366 
367  TRY {
368  dv_new(t);

371  dv_zero(_b, FB_DIGS + 1);
372 
373  fb_copy(_b, b);
374 
375  for (int i = 0; i < FB_DIGIT; i++) {
376  for (int j = 0; j < FB_DIGS; j++) {
377  if (a[j] & ((dig_t)1 << i)) {
378  fb_addd_low(t + j, t + j, _b, FB_DIGS + 1);
379  }
380  }
381  if (i != FB_DIGIT - 1) {
382  carry = fb_lsh1_low(_b, _b);
383  _b[FB_DIGS] = (_b[FB_DIGS] << 1) | carry;
384  }
385  }
386

388  }
389  CATCH_ANY {
