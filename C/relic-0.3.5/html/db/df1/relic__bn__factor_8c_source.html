38 int bn_factor(bn_t c, const bn_t a) {
39  bn_t t0, t1;
40  int result;
41  unsigned int i, tests;
42 
43  bn_null(t0);
44  bn_null(t1);
45 
46  result = 1;
47 
48  if (bn_is_even(a)) {
49  bn_set_dig(c, 2);
50  return 1;
51  }
52 
53  TRY {
54  bn_new(t0);
55  bn_new(t1);
56 
57  bn_set_dig(t0, 2);
58 
59 #if WORD == 8
60  tests = 255;
61 #else
62  tests = 65535;
63 #endif
64  for (i = 2; i < tests; i++) {
65  bn_set_dig(t1, i);
66  bn_mxp(t0, t0, t1, a);
67  }
68 
69  bn_sub_dig(t0, t0, 1);
70  bn_gcd(t1, t0, a);
71  if (bn_cmp_dig(t1, 1) == CMP_GT && bn_cmp(t1, a) == CMP_LT) {
72  bn_copy(c, t1);
73  } else {
74  result = 0;
75  }
76  } CATCH_ANY {
77  THROW(ERR_CAUGHT);
78  } FINALLY {
79  bn_free(t0);
80  bn_free(t1);
81  }
82  return result;
83 }
