49 static void ep2_dbl_basic_imp(ep2_t r, fp2_t s, ep2_t p) {
50  fp2_t t0, t1, t2;
51 
52  fp2_null(t0);
53  fp2_null(t1);
54  fp2_null(t2);
55 
56  TRY {
57  fp2_new(t0);
58  fp2_new(t1);
59  fp2_new(t2);
60 
61  /* t0 = 1/(2 * y1). */
62  fp2_dbl(t0, p->y);
63  fp2_inv(t0, t0);
64 
65  /* t1 = 3 * x1^2 + a. */
66  fp2_sqr(t1, p->x);
67  fp2_copy(t2, t1);
68  fp2_dbl(t1, t1);
69  fp2_add(t1, t1, t2);
70 
71  if (ep2_curve_is_twist()) {
72  switch (ep_curve_opt_a()) {
73  case OPT_ZERO:
74  break;
75  case OPT_ONE:
76  fp_set_dig(t2[0], 1);
77  fp2_mul_art(t2, t2);
78  fp2_mul_art(t2, t2);
79  fp2_add(t1, t1, t2);
80  break;
81  case OPT_DIGIT:
82  fp_set_dig(t2[0], ep_curve_get_a()[0]);
83  fp2_mul_art(t2, t2);
84  fp2_mul_art(t2, t2);
85  fp2_add(t1, t1, t2);
86  break;
87  default:
88  fp_copy(t2[0], ep_curve_get_a());
89  fp_zero(t2[1]);
90  fp2_mul_art(t2, t2);
91  fp2_mul_art(t2, t2);
92  fp2_add(t1, t1, t2);
93  break;
94  }
95  }
96 
97  /* t1 = (3 * x1^2 + a)/(2 * y1). */
98  fp2_mul(t1, t1, t0);
99 
100  if (s != NULL) {
101  fp2_copy(s, t1);
102  }
103 
104  /* t2 = t1^2. */
105  fp2_sqr(t2, t1);
106 
107  /* x3 = t1^2 - 2 * x1. */
108  fp2_dbl(t0, p->x);
109  fp2_sub(t0, t2, t0);
110 
111  /* y3 = t1 * (x1 - x3) - y1. */
112  fp2_sub(t2, p->x, t0);
113  fp2_mul(t1, t1, t2);
114 
115  fp2_sub(r->y, t1, p->y);
116 
117  fp2_copy(r->x, t0);
118  fp2_copy(r->z, p->z);
119 
120  r->norm = 1;
121  }
122  CATCH_ANY {
123  THROW(ERR_CAUGHT);
124  }
125  FINALLY {
126  fp2_free(t0);
127  fp2_free(t1);
128  fp2_free(t2);
129  }
130 }

163  fp2_add(t1, t1, t1);
164  fp2_add(t1, t1, t1);

167  fp2_sub(r->x, r->x, t2);
168  fp2_mul(r->z, p->z, p->y);

171 
172  fp2_sqr(t3, t3);
173  fp2_add(t3, t3, t3);
174  fp2_sub(t1, t1, r->x);
175  fp2_mul(r->y, t0, t1);
176  fp2_sub(r->y, r->y, t3);
177

179  }
