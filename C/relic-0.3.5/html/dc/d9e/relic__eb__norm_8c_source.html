52 static void eb_norm_ordin(eb_t r, const eb_t p, int flag) {
53  if (!p->norm) {
54  if (flag) {
55  fb_copy(r->z, p->z);
56  } else {
57  fb_inv(r->z, p->z);
58  }
59  fb_mul(r->x, p->x, r->z);
60  fb_sqr(r->z, r->z);
61  fb_mul(r->y, p->y, r->z);
62  fb_set_dig(r->z, 1);

114 void eb_norm(eb_t r, const eb_t p) {
115  if (eb_is_infty(p)) {
116  eb_set_infty(r);
117  return;
118  }
119 
120  if (p->norm == 1) {
121  /* If the point is represented in affine coordinates, we just copy it. */
122  eb_copy(r, p);
123  return;
124  }
125 
126  if (p->norm == 2) {
127  eb_norm_halve(r, p);
128  return;
129  }
130 
131 #if EB_ADD == PROJC || !defined(STRIP)
132 
133 #if defined(EB_SUPER)
134  if (eb_curve_is_super()) {
135  eb_norm_super(r, p, 0);
136  return;
137  }
138 #endif /* EB_SUPER */
139 
140 #if defined(EB_ORDIN) || defined(EB_KBLTZ)
141  eb_norm_ordin(r, p, 0);
142 #endif /* EB_ORDIN || EB_KBLTZ */
143 
144 #endif /* EB_ADD == PROJC */
145 }

147 void eb_norm_sim(eb_t *r, const eb_t *t, int n) {
148  int i;
149  fb_t a[n];
150 
151  if (n == 1) {
152  eb_norm(r[0], t[0]);
153  return;
154  }
155 
156  for (i = 0; i < n; i++) {
157  fb_null(a[i]);
158  }
159 
160  TRY {
161  for (i = 0; i < n; i++) {
162  fb_new(a[i]);
163  if (!eb_is_infty(t[i])) {
164  fb_copy(a[i], t[i]->z);
165  } else {
166  fb_set_dig(a[i], 1);
167  }
168  }
169 
170  fb_inv_sim(a, (const fb_t *)a, n);
171 
172  for (i = 0; i < n; i++) {
173  fb_copy(r[i]->x, t[i]->x);
174  fb_copy(r[i]->y, t[i]->y);
175  if (!eb_is_infty(t[i])) {
176  fb_copy(r[i]->z, a[i]);
177  }
178  }
179  }
180  CATCH_ANY {
181  THROW(ERR_CAUGHT);
182  }
183  FINALLY {
184  for (i = 0; i < n; i++) {
185  fb_free(a[i]);
186  }
187  }
188 
189 #if defined(EB_SUPER)
190  if (eb_curve_is_super()) {
191  for (i = 0; i < n; i++) {
192  eb_norm_super(r[i], r[i], 1);
193  }
194  }
195 #endif
196 #if defined(EB_ORDIN) || defined(EB_KBLTZ)
197  if (!eb_curve_is_super()) {
198  for (i = 0; i < n; i++) {
199  eb_norm_ordin(r[i], r[i], 1);
200  }
201  }
202 #endif
203 }
