41 void eb_pck(eb_t r, const eb_t p) {
42  if (eb_curve_is_super()) {
43  /* z3 = y1/c. */
44  fb_inv(r->z, eb_curve_get_c());
45  fb_mul(r->z, r->z, p->y);
46  /* x3 = x1. */
47  fb_copy(r->x, p->x);
48  /* y3 = b(y1/c). */
49  fb_set_dig(r->y, fb_get_bit(r->z, 0));
50  /* z3 = 1. */
51  fb_set_dig(r->z, 1);
52  } else {
53  /* z3 = y1/x1. */
54  fb_inv(r->z, p->x);
55  fb_mul(r->z, r->z, p->y);
56  /* x3 = x1. */
57  fb_copy(r->x, p->x);
58  /* y3 = b(y1/x1). */
59  fb_set_dig(r->y, fb_get_bit(r->z, 0));
60  /* z3 = 1. */
61  fb_set_dig(r->z, 1);
62  }
63 
64  r->norm = 1;
65 }

67 int eb_upk(eb_t r, const eb_t p) {
68  fb_t t0, t1;
69  int res = 0;
70 
71  fb_null(t0);
72  fb_null(t1);
73 
74  TRY {
75  fb_new(t0);
76  fb_new(t1);
77 
78  eb_rhs(t1, p);
79 
80  if (eb_curve_is_super()) {
81  /* t0 = c^2. */
82  fb_sqr(t0, eb_curve_get_c());
83  /* t0 = 1/c^2. */
84  fb_inv(t0, t0);
85  /* t0 = t1/c^2. */
86  fb_mul(t0, t0, t1);
87  res = (fb_trc(t0) == 0);
88  /* Solve t1^2 + t1 = t0. */
89  fb_slv(t1, t0);
90  /* If this is not the correct solution, try the other. */
91  if (fb_get_bit(t1, 0) != fb_get_bit(p->y, 0)) {
92  fb_add_dig(t1, t1, 1);
93  }
94  /* x3 = x1, y3 = t1 * c, z3 = 1. */
95  fb_mul(r->y, t1, eb_curve_get_c());
96  } else {
97  fb_sqr(t0, p->x);
98  /* t0 = 1/x1^2. */
99  fb_inv(t0, t0);
100  /* t0 = t1/x1^2. */
101  fb_mul(t0, t0, t1);
102  res = (fb_trc(t0) == 0);
103  /* Solve t1^2 + t1 = t0. */
104  fb_slv(t1, t0);
105  /* If this is not the correct solution, try the other. */
106  if (fb_get_bit(t1, 0) != fb_get_bit(p->y, 0)) {
107  fb_add_dig(t1, t1, 1);
108  }
109  /* x3 = x1, y3 = t1 * x1, z3 = 1. */
110  fb_mul(r->y, t1, p->x);
111  }
112  fb_copy(r->x, p->x);
113  fb_set_dig(r->z, 1);
114 
115  r->norm = 1;
116  }
117  CATCH_ANY {
118  THROW(ERR_CAUGHT);
119  }
120  FINALLY {
121  fb_free(t0);
122  fb_free(t1);
123  }
124  return res;
125 }
