40 void fb2_mul(fb2_t c, fb2_t a, fb2_t b) {
41  fb_t t0, t1, t2;
42 
43  fb_null(t0);
44  fb_null(t1);
45  fb_null(t2);
46 
47  TRY {
48  fb_new(t0);
49  fb_new(t1);
50  fb_new(t2);
51 
52  fb_add(t0, a[0], a[1]);
53  fb_add(t1, b[0], b[1]);
54 
55  fb_mul(t0, t0, t1);
56  fb_mul(t1, a[0], b[0]);
57  fb_mul(t2, a[1], b[1]);
58 
59  fb_add(c[0], t1, t2);
60  fb_add(c[1], t0, t1);
61  } CATCH_ANY {
62  THROW(ERR_CAUGHT);
63  } FINALLY {
64  fb_free(t0);
65  fb_free(t1);
66  fb_free(t2);
67  }
68 }
