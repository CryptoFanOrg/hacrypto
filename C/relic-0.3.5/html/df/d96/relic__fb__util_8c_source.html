193 void fb_size(int *size, const fb_t a, int radix) {
194  int digits = 0, l;
195  fb_t t;
196 
197  fb_null(t);
198 
199  *size = 0;
200 
201  if (fb_is_zero(a)) {
202  *size = 2;
203  return;
204  }
205 
206  /* Binary case. */
207  if (radix == 2) {
208  *size = fb_bits(a) + 1;
209  return;
210  }
211 
212  /* Check the radix. */
213  if (!valid_radix(radix)) {
214  THROW(ERR_NO_VALID);
215  }
216  l = log_radix(radix);
217 
218  TRY {
219  fb_new(t);
220 
221  fb_copy(t, a);
222  digits = 0;
223  while (fb_is_zero(t) == 0) {
224  fb_rsh(t, t, l);
225  digits++;
226  }
227  }
228  CATCH_ANY {
229  THROW(ERR_CAUGHT);
230  }
231  FINALLY {
232  fb_free(t);
233  }
234 
235  *size = digits + 1;
236 }

238 void fb_read(fb_t a, const char *str, int len, int radix) {
239  int i, j, l;
240  char c;
241  dig_t carry;
242 
243  fb_zero(a);
244 
245  if (!valid_radix(radix)) {
246  THROW(ERR_NO_VALID);
247  }
248  l = log_radix(radix);
249 
250  j = 0;
251  while (str[j] && j < len) {
252  c = (char)((radix < 36) ? TOUPPER(str[j]) : str[j]);
253  for (i = 0; i < 64; i++) {
254  if (c == util_conv_char(i)) {
255  break;
256  }
257  }
258 
259  if (i < radix) {
260  carry = fb_lshb_low(a, a, l);
261  if (carry != 0) {
262  THROW(ERR_NO_BUFFER);
263  }
264  fb_add_dig(a, a, (dig_t)i);
265  } else {
266  break;
267  }
268  j++;
269  }
270 }

272 void fb_write(char *str, int len, const fb_t a, int radix) {
273  fb_t t;
274  int d, l, i, j;
275  char c;
276 
277  fb_null(t);
278 
279  fb_size(&l, a, radix);
280  if (len <= l) {
281  THROW(ERR_NO_BUFFER);
282  }
283  len = l;
284 
285  l = log_radix(radix);
286  if (!valid_radix(radix)) {
287  THROW(ERR_NO_VALID)
288  }
289 
290  if (fb_is_zero(a) == 1) {
291  *str++ = '0';
292  *str = '\0';
293  return;
294  }
295 
296  TRY {
297  fb_new(t);
298  fb_copy(t, a);
299 
300  j = 0;
301  while (!fb_is_zero(t)) {
302  d = t[0] % radix;
303  fb_rshb_low(t, t, l);
304  str[j] = util_conv_char(d);
305  j++;
306  }
307 
308  /* Reverse the digits of the string. */
309  i = 0;
310  j = len - 2;
311  while (i < j) {
312  c = str[i];
313  str[i] = str[j];
314  str[j] = c;
315  ++i;
316  --j;
317  }
318 
319  str[len - 1] = '\0';
320  }
321  CATCH_ANY {
322  THROW(ERR_CAUGHT);
323  }
324  FINALLY {
325  fb_free(t);
326  }
327 }
