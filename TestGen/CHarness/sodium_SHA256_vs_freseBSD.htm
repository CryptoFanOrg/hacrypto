     state->count[0] += bitlen[0];  
   
   /* Finish the current block */     if (inlen < 64 - r) { 
   memcpy(&ctx->buf[r], src, 64 - r);         memcpy(&state->buf[r], in, inlen); 
   SHA256_Transform(ctx->state, ctx->buf);         return 0; 
   src += 64 - r;     } 
   len -= 64 - r;     memcpy(&state->buf[r], in, 64 - r); 
     SHA256_Transform(state->state, state->buf);  
   /* Perform complete blocks */     in += 64 - r; 
   while (len >= 64) {     inlen -= 64 - r; 
       SHA256_Transform(ctx->state, src);    
       src += 64;     while (inlen >= 64) { 
       len -= 64;         SHA256_Transform(state->state, in); 
         in += 64;  
         inlen -= 64;  
   }    }
     memcpy(state->buf, in, inlen);  
   
   /* Copy left over data into buffer */     return 0; 
   memcpy(ctx->buf, src, len);    
} }
   
/*  int 
* SHA-256 finalization.  Pads the input data, exports the hash value,  crypto_hash_sha256_final(crypto_hash_sha256_state *state, 
* and clears the context state.                          unsigned char *out) 
*/    
void    
SHA256_Final(unsigned char digest[32], SHA256_CTX * ctx)    
{ {
     SHA256_Pad(state);  
     be32enc_vect(out, state->state, 32);  
     sodium_memzero((void *) state, sizeof *state);  
   
   /* Add padding */     return 0; 
   SHA256_Pad(ctx);  } 
   
  int  
  crypto_hash(unsigned char *out, const unsigned char *in,  
             unsigned long long inlen)  
  {  
     crypto_hash_sha256_state state;  
   
   /* Write the hash */     crypto_hash_sha256_init(&state); 
   be32enc_vect(digest, ctx->state, 32);     crypto_hash_sha256_update(&state, in, inlen); 
     crypto_hash_sha256_final(&state, out);  
   
   /* Clear the context state */     return 0; 
   memset((void *)ctx, 0, sizeof(*ctx));    
} }
