39 void ep_map(ep_t p, const unsigned char *msg, int len) {
40  bn_t k;
41  fp_t t;
42  unsigned char digest[MD_LEN];
43 
44  bn_null(k);
45  fp_null(t);
46 
47  TRY {
48  bn_new(k);
49  fp_new(t);
50 
51  md_map(digest, msg, len);
52  bn_read_bin(k, digest, MIN(FP_BYTES, MD_LEN));
53 
54  fp_prime_conv(p->x, k);
55  fp_zero(p->y);
56  fp_set_dig(p->z, 1);
57 
58  while (1) {
59  ep_rhs(t, p);
60 
61  if (fp_srt(p->y, t)) {
62  p->norm = 1;
63  break;
64  }
65  fp_add_dig(p->x, p->x, 1);
66  }
67 
68  /* Now, multiply by cofactor to get the correct group. */
69  ep_curve_get_cof(k);
70  if (bn_bits(k) < BN_DIGIT) {
71  ep_mul_dig(p, p, k->dp[0]);
72  } else {
73  ep_mul(p, p, k);
74  }
75  }
76  CATCH_ANY {
77  THROW(ERR_CAUGHT);
78  }
79  FINALLY {
80  bn_free(k);
81  fp_free(t);
82  }
83 }
