3 /*
4  * Description:
5  * This file implements the Secure Hash Signature Standard
6  * algorithms as defined in the National Institute of Standards
7  * and Technology Federal Information Processing Standards
8  * Publication (FIPS PUB) 180-1 published on April 17, 1995, 180-2
9  * published on August 1, 2002, and the FIPS PUB 180-2 Change
10  * Notice published on February 28, 2004.
11  *
12  * A combined document showing all algorithms is available at
13  * http://csrc.nist.gov/publications/fips/
14  * fips180-2/fips180-2withchangenotice.pdf
15  *
16  * The SHA-1 algorithm produces a 160-bit message digest for a
17  * given data stream. It should take about 2**n steps to find a
18  * message with the same digest as a given message and
19  * 2**(n/2) to find any two messages with the same digest,
20  * when n is the digest size in bits. Therefore, this
21  * algorithm can serve as a means of providing a
22  * "fingerprint" for a message.
23  *
24  * Portability Issues:
25  * SHA-1 is defined in terms of 32-bit "words". This code
26  * uses <stdint.h> (included via "sha.h") to define 32 and 8
27  * bit unsigned integer types. If your C compiler does not
28  * support 32 bit unsigned integers, this code is not
29  * appropriate.
30  *
31  * Caveats:
32  * SHA-1 is designed to work with messages less than 2^64 bits
33  * long. This implementation uses SHA1Input() to hash the bits
34  * that are a multiple of the size of an 8-bit character, and then
35  * uses SHA1FinalBits() to hash the final few bits of the input.
36  */

97  * SHA1Input
98  *
99  * Description:
100  * This function accepts an array of octets as the next portion
101  * of the message.
102  *
103  * Parameters:
104  * context: [in/out]
105  * The SHA context to update
106  * message_array: [in]
107  * An array of characters representing the next portion of
108  * the message.
109  * length: [in]
110  * The length of the message in message_array
111  *
112  * Returns:
113  * sha Error Code.
114  *
115  */
116 int SHA1Input(SHA1Context *context,
117  const uint8_t *message_array, unsigned length)
118 {
119  if (!length)
120  return shaSuccess;
121 
122  if (!context || !message_array)
123  return shaNull;
124 
125  if (context->Computed) {
126  context->Corrupted = shaStateError;
127  return shaStateError;
128  }
129  if (context->Corrupted)
130  return context->Corrupted;

133  context->Message_Block[context->Message_Block_Index++] =
134  (unsigned char)(*message_array & 0xFF);

139  SHA1ProcessMessageBlock(context);

147 /*
148  * SHA1FinalBits
149  *
150  * Description:
151  * This function will add in any final bits of the message.
152  *
153  * Parameters:
154  * context: [in/out]
155  * The SHA context to update
156  * message_bits: [in]
157  * The final bits of the message, in the upper portion of the
158  * byte. (Use 0b###00000 instead of 0b00000### to input the
159  * three bits ###.)
160  * length: [in]
161  * The number of bits in message_bits, between 1 and 7.
162  *
163  * Returns:
164  * sha Error Code.
165  */
166 int SHA1FinalBits(SHA1Context *context, const uint8_t message_bits,
167  unsigned int length)
168 {
169  uint8_t masks[8] = {
170  /* 0 0b00000000 */ 0x00, /* 1 0b10000000 */ 0x80,
171  /* 2 0b11000000 */ 0xC0, /* 3 0b11100000 */ 0xE0,
172  /* 4 0b11110000 */ 0xF0, /* 5 0b11111000 */ 0xF8,
173  /* 6 0b11111100 */ 0xFC, /* 7 0b11111110 */ 0xFE
174  };
175  uint8_t markbit[8] = {
176  /* 0 0b10000000 */ 0x80, /* 1 0b01000000 */ 0x40,
177  /* 2 0b00100000 */ 0x20, /* 3 0b00010000 */ 0x10,
178  /* 4 0b00001000 */ 0x08, /* 5 0b00000100 */ 0x04,
179  /* 6 0b00000010 */ 0x02, /* 7 0b00000001 */ 0x01
180  };
181 
182  if (!length)
183  return shaSuccess;
184 
185  if (!context)
186  return shaNull;
187 
188  if (context->Computed || (length >= 8) || (length == 0)) {
189  context->Corrupted = shaStateError;
190  return shaStateError;
191  }
192 
193  if (context->Corrupted)
194  return context->Corrupted;
195 
196  uint32_t addTemp;
197  SHA1AddLength(context, length);
198  SHA1Finalize(context,
199  (uint8_t) ((message_bits & masks[length]) | markbit[length]));
200 
201  return shaSuccess;
202 }

212  *
213  * Parameters:

218  *

221  *
222  */

224  uint8_t Message_Digest[SHA1HashSize])
225 {
226  int i;
227  if (!context || !Message_Digest)
228  return shaNull;
229 
230  if (context->Corrupted)
231  return context->Corrupted;
232 
233  if (!context->Computed)
234  SHA1Finalize(context, 0x80);
235 
236  for (i = 0; i < SHA1HashSize; ++i)
237  Message_Digest[i] = (uint8_t) (context->Intermediate_Hash[i>>2]
238  >> 8 * ( 3 - ( i & 0x03 ) ));

244  * SHA1Finalize
245  *
246  * Description:
247  * This helper function finishes off the digest calculations.
248  *
249  * Parameters:
250  * context: [in/out]
251  * The SHA context to update
252  * Pad_Byte: [in]
253  * The last byte to add to the digest before the 0-padding
254  * and length. This will contain the last bits of the message
255  * followed by another single bit. If the message was an
256  * exact multiple of 8-bits long, Pad_Byte will be 0x80.
257  *
258  * Returns:
259  * sha Error Code.
260  *
261  */
262 static void SHA1Finalize(SHA1Context *context, uint8_t Pad_Byte)
263 {
264  int i;
265  SHA1PadMessage(context, Pad_Byte);
266  /* message may be sensitive, clear it out */
267  for (i = 0; i < SHA1_Message_Block_Size; ++i)
268  context->Message_Block[i] = 0;
269  context->Length_Low = 0; /* and clear length */
270  context->Length_High = 0;
271  context->Computed = 1;

274 /*
275  * SHA1PadMessage
276  *
277  * Description:
278  * According to the standard, the message must be padded to an
279  * even 512 bits. The first padding bit must be a '1'. The last
280  * 64 bits represent the length of the original message. All bits
281  * in between should be 0. This helper function will pad the
282  * message according to those rules by filling the Message_Block
283  * array accordingly. When it returns, it can be assumed that the
284  * message digest has been computed.
285  *
286  * Parameters:
287  * context: [in/out]
288  * The context to pad
289  * Pad_Byte: [in]
290  * The last byte to add to the digest before the 0-padding
291  * and length. This will contain the last bits of the message
292  * followed by another single bit. If the message was an
293  * exact multiple of 8-bits long, Pad_Byte will be 0x80.
294  *
295  * Returns:
296  * Nothing.
297  */
298 static void SHA1PadMessage(SHA1Context *context, uint8_t Pad_Byte)
299 {
300  /*
301  * Check to see if the current message block is too small to hold
302  * the initial padding bits and length. If so, we will pad the
303  * block, process it, and then continue padding into a second
304  * block.
305  */
306  if (context->Message_Block_Index >= (SHA1_Message_Block_Size - 8)) {
307  context->Message_Block[context->Message_Block_Index++] = Pad_Byte;
308  while (context->Message_Block_Index < SHA1_Message_Block_Size)
309  context->Message_Block[context->Message_Block_Index++] = 0;
310 
311  SHA1ProcessMessageBlock(context);
312  } else
313  context->Message_Block[context->Message_Block_Index++] = Pad_Byte;
314 
315  while (context->Message_Block_Index < (SHA1_Message_Block_Size - 8))
316  context->Message_Block[context->Message_Block_Index++] = 0;
317 
318  /*
319  * Store the message length as the last 8 octets
320  */
321  context->Message_Block[56] = (uint8_t) (context->Length_High >> 24);
322  context->Message_Block[57] = (uint8_t) (context->Length_High >> 16);
323  context->Message_Block[58] = (uint8_t) (context->Length_High >> 8);
324  context->Message_Block[59] = (uint8_t) (context->Length_High);
325  context->Message_Block[60] = (uint8_t) (context->Length_Low >> 24);
326  context->Message_Block[61] = (uint8_t) (context->Length_Low >> 16);
327  context->Message_Block[62] = (uint8_t) (context->Length_Low >> 8);
328  context->Message_Block[63] = (uint8_t) (context->Length_Low);
329 
330  SHA1ProcessMessageBlock(context);
331 }

334  * SHA1ProcessMessageBlock
335  *
336  * Description:
337  * This helper function will process the next 512 bits of the
338  * message stored in the Message_Block array.
339  *
340  * Parameters:
341  * None.
342  *
343  * Returns:
344  * Nothing.
345  *
346  * Comments:
347  * Many of the variable names in this code, especially the
348  * single character names, were used because those were the
349  * names used in the publication.
350  */
351 static void SHA1ProcessMessageBlock(SHA1Context *context)
352 {
353  /* Constants defined in FIPS-180-2, section 4.2.1 */
354  const uint32_t K[4] = {
355  0x5A827999, 0x6ED9EBA1, 0x8F1BBCDC, 0xCA62C1D6
356  };
357  int t; /* Loop counter */
358  uint32_t temp; /* Temporary word value */
359  uint32_t W[80]; /* Word sequence */
360  uint32_t A, B, C, D, E; /* Word buffers */
361 
362  /*
363  * Initialize the first 16 words in the array W
364  */
365  for (t = 0; t < 16; t++) {
366  W[t] = ((uint32_t)context->Message_Block[t * 4]) << 24;
367  W[t] |= ((uint32_t)context->Message_Block[t * 4 + 1]) << 16;
368  W[t] |= ((uint32_t)context->Message_Block[t * 4 + 2]) << 8;
369  W[t] |= ((uint32_t)context->Message_Block[t * 4 + 3]);
370  }
371 
372  for (t = 16; t < 80; t++)
373  W[t] = SHA1_ROTL(1, W[t-3] ^ W[t-8] ^ W[t-14] ^ W[t-16]);
374 
375  A = context->Intermediate_Hash[0];
376  B = context->Intermediate_Hash[1];
377  C = context->Intermediate_Hash[2];
378  D = context->Intermediate_Hash[3];
379  E = context->Intermediate_Hash[4];
380 
381  for (t = 0; t < 20; t++) {
382  temp = SHA1_ROTL(5,A) + SHA_Ch(B, C, D) + E + W[t] + K[0];
383  E = D;
384  D = C;
385  C = SHA1_ROTL(30,B);
386  B = A;
387  A = temp;
388  }
389 
390  for (t = 20; t < 40; t++) {
391  temp = SHA1_ROTL(5,A) + SHA_Parity(B, C, D) + E + W[t] + K[1];
392  E = D;
393  D = C;
394  C = SHA1_ROTL(30,B);
395  B = A;
396  A = temp;
397  }
398 
399  for (t = 40; t < 60; t++) {
400  temp = SHA1_ROTL(5,A) + SHA_Maj(B, C, D) + E + W[t] + K[2];
401  E = D;
402  D = C;
403  C = SHA1_ROTL(30,B);
404  B = A;
405  A = temp;
406  }
407 
408  for (t = 60; t < 80; t++) {
409  temp = SHA1_ROTL(5,A) + SHA_Parity(B, C, D) + E + W[t] + K[3];
410  E = D;
411  D = C;
412  C = SHA1_ROTL(30,B);
413  B = A;
414  A = temp;
415  }
416 
417  context->Intermediate_Hash[0] += A;
418  context->Intermediate_Hash[1] += B;
419  context->Intermediate_Hash[2] += C;
420  context->Intermediate_Hash[3] += D;
421  context->Intermediate_Hash[4] += E;
422 
423  context->Message_Block_Index = 0;
424 }
