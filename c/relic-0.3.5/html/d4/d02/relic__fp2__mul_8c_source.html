40 void fp2_mul_frb(fp2_t c, fp2_t a, int i, int j) {
41  ctx_t *ctx = core_get();
42 
43  if (i == 2) {
44  fp_mul(c[0], a[0], ctx->fp2_p2[j - 1]);
45  fp_mul(c[1], a[1], ctx->fp2_p2[j - 1]);
46  } else {
47 #if ALLOC == AUTO
48  if (i == 1) {

81 void fp2_mul_basic(fp2_t c, fp2_t a, fp2_t b) {
82  dv_t t0, t1, t2, t3, t4;
83 
84  dv_null(t0);
85  dv_null(t1);
86  dv_null(t2);
87  dv_null(t3);
88  dv_null(t4);
89 
90  TRY {
91  dv_new(t0);
92  dv_new(t1);
93  dv_new(t2);
94  dv_new(t3);
95  dv_new(t4);
96 
97  /* Karatsuba algorithm. */
98 
99  /* t2 = a_0 + a_1, t1 = b0 + b1. */
100  fp_add(t2, a[0], a[1]);
101  fp_add(t1, b[0], b[1]);
102 
103  /* t3 = (a_0 + a_1) * (b0 + b1). */
104  fp_muln_low(t3, t2, t1);
105 
106  /* t0 = a_0 * b0, t4 = a_1 * b1. */
107  fp_muln_low(t0, a[0], b[0]);
108  fp_muln_low(t4, a[1], b[1]);
109 
110  /* t2 = (a_0 * b0) + (a_1 * b1). */
111  fp_addc_low(t2, t0, t4);
112 
113  /* t1 = (a_0 * b0) + u^2 * (a_1 * b1). */
114  fp_subc_low(t1, t0, t4);
115 
116  /* t1 = u^2 * (a_1 * b1). */
117  for (int i = -1; i > fp_prime_get_qnr(); i--) {
118  fp_subc_low(t1, t1, t4);
119  }
120  /* c_0 = t1 mod p. */
121  fp_rdc(c[0], t1);
122 
123  /* t4 = t3 - t2. */
124  fp_subc_low(t4, t3, t2);
125 
126  /* c_1 = t4 mod p. */
127  fp_rdc(c[1], t4);
128  }
129  CATCH_ANY {
130  THROW(ERR_CAUGHT);
131  }
132  FINALLY {
133  dv_free(t0);
134  dv_free(t1);
135  dv_free(t2);
136  dv_free(t3);
137  dv_free(t4);
138  }
139 }

141 void fp2_mul_nor_basic(fp2_t c, fp2_t a) {
142  fp2_t t;
143  bn_t b;
144 
145  fp2_null(t);
146  bn_null(b);
147 
148  TRY {
149  fp2_new(t);
150  bn_new(b);
151 
152 #ifdef FP_QNRES
153  /* If p = 3 mod 8, (1 + i) is a QNR/CNR. */
154  fp_neg(t[0], a[1]);
155  fp_add(c[1], a[0], a[1]);
156  fp_add(c[0], t[0], a[0]);
157 #else
158  switch (fp_prime_get_mod8()) {
159  case 3:
160  /* If p = 3 mod 8, (1 + u) is a QNR/CNR. */
161  fp_neg(t[0], a[1]);
162  fp_add(c[1], a[0], a[1]);
163  fp_add(c[0], t[0], a[0]);
164  break;
165  case 1:
166  case 5:
167  /* If p = 5 mod 8, (u) is a QNR/CNR. */
168  fp2_mul_art(c, a);
169  break;
170  case 7:
171  /* If p = 7 mod 8, we choose (4 + u) is a QNR/CNR. */
172  fp2_mul_art(t, a);
173  fp2_dbl(c, a);
174  fp2_dbl(c, c);
175  fp2_add(c, c, t);
176  break;
177  default:
178  THROW(ERR_NO_VALID);
179  }
180 #endif
181  }
182  CATCH_ANY {
183  THROW(ERR_CAUGHT);
184  }
185  FINALLY {
186  fp2_free(t);
187  bn_free(b);
188  }
189 }

205 void fp2_mul_art(fp2_t c, fp2_t a) {
206  fp_t t;
207 
208  fp_null(t);
209 
210  TRY {
211  fp_new(t);
212 
213 #ifdef FP_QNRES
214  /* (a_0 + a_1 * i) * i = -a_1 + a_0 * i. */
215  fp_copy(t, a[0]);
216  fp_neg(c[0], a[1]);
217  fp_copy(c[1], t);
218 #else
219  /* (a_0 + a_1 * u) * u = (a_1 * u^2) + a_0 * u. */
220  fp_copy(t, a[0]);
221  fp_neg(c[0], a[1]);
222  for (int i = -1; i > fp_prime_get_qnr(); i--) {
223  fp_sub(c[0], c[0], a[1]);
224  }
225  fp_copy(c[1], t);
226 #endif
227  }
228  CATCH_ANY {
229  THROW(ERR_CAUGHT);
230  }
231  FINALLY {
232  fp_free(t);
233  }
234 }
