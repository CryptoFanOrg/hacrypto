41 void eb_hlv(eb_t r, const eb_t p) {
42  fb_t l, t;
43 
44  fb_null(l);
45  fb_null(t);
46 
47  TRY {
48  fb_new(l);
49  fb_new(t);
50 
51  /* Solve l^2 + l = u + a. */
52  switch (eb_curve_opt_a()) {
53  case OPT_ZERO:
54  fb_copy(t, p->x);
55  break;
56  case OPT_ONE:
57  fb_add_dig(t, p->x, (dig_t)1);
58  break;
59  case OPT_DIGIT:
60  fb_add_dig(t, p->x, eb_curve_get_a()[0]);
61  break;
62  default:
63  fb_add(t, p->x, eb_curve_get_a());
64  break;
65  }
66 
67  fb_slv(l, t);
68 
69  if (p->norm == 1) {
70  /* Compute t = v + u * lambda. */
71  fb_mul(t, l, p->x);
72  fb_add(t, t, p->y);
73  } else {
74  /* Compute t = u * (u + lambda_P + lambda). */
75  fb_add(t, l, p->y);
76  fb_add(t, t, p->x);
77  fb_mul(t, t, p->x);
78  }
79 
80  /* If Tr(t) = 0 then lambda_P = lambda, u = sqrt(t + u). */
81  if (fb_trc(t) == 0) {
82  fb_copy(r->y, l);
83  fb_add(t, t, p->x);
84  fb_srt(r->x, t);
85  } else {
86  /* Else lambda_P = lambda + 1, u = sqrt(t). */
87  fb_add_dig(r->y, l, 1);
88  fb_srt(r->x, t);
89  }
90  fb_set_dig(r->z, 1);
91  r->norm = 2;
92  }
93  CATCH_ANY {
94  THROW(ERR_CAUGHT);
95  }
96  FINALLY {
97  fb_free(l);
98  fb_free(t);
99  }
100 }
