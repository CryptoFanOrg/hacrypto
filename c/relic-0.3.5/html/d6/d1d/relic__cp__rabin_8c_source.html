60 int cp_rabin_gen(rabin_t pub, rabin_t prv, int bits) {
61  bn_t r;
62  int result = STS_OK;
63 
64  bn_null(r);
65 
66  TRY {
67  bn_new(r);
68 
69  /* Generate different primes p and q. */
70  do {
71  bn_gen_prime(prv->p, bits / 2);
72  bn_mod_2b(r, prv->p, 2);
73  } while (bn_cmp_dig(r, 3) != CMP_EQ);
74 
75  do {
76  bn_gen_prime(prv->q, bits / 2);
77  bn_mod_2b(r, prv->q, 2);
78  } while (bn_cmp(prv->p, prv->q) == CMP_EQ || bn_cmp_dig(r, 3) != CMP_EQ);
79 
80  /* Swap p and q so that p is smaller. */
81  if (bn_cmp(prv->p, prv->q) == CMP_LT) {
82  bn_copy(r, prv->p);
83  bn_copy(prv->p, prv->q);
84  bn_copy(prv->q, r);
85  }
86 
87  bn_gcd_ext(r, prv->dp, prv->dq, prv->p, prv->q);
88  if (bn_cmp_dig(r, 1) != CMP_EQ) {
89  result = STS_ERR;
90  }
91 
92  bn_mul(prv->n, prv->p, prv->q);
93  bn_copy(pub->n, prv->n);
94  }
95  CATCH_ANY {
96  result = STS_ERR;
97  }
98  FINALLY {
99  bn_free(r);
100  }
101 
102  return result;
103 }

105 int cp_rabin_enc(unsigned char *out, int *out_len, unsigned char *in, int in_len,
106  rabin_t pub) {
107  bn_t m, t;
108  int size, result = STS_OK;
109 
110  bn_null(m);
111  bn_null(t);
112 
113  bn_size_bin(&size, pub->n);
114 
115  if (in_len > (size - RABIN_PAD_LEN - 2)) {
116  return STS_ERR;
117  }
118 
119  TRY {
120  bn_new(m);
121  bn_new(t);
122  bn_zero(m);
123 
124  /* EB = 00 | FF || D || SUFFIX */
125  bn_lsh(m, m, 8);
126  bn_add_dig(m, m, RABIN_PAD);
127  bn_lsh(m, m, in_len * 8);
128  bn_read_bin(t, in, in_len);
129  bn_add(m, m, t);
130  bn_mod_2b(t, m, 8 * RABIN_PAD_LEN);
131  bn_lsh(m, m, 8 * RABIN_PAD_LEN);
132  bn_add(m, m, t);
133 
134  bn_sqr(m, m);
135  bn_mod(m, m, pub->n);
136 
137  if (size <= *out_len) {
138  *out_len = size;
139  memset(out, 0, *out_len);
140  bn_write_bin(out, size, m);
141  } else {
142  result = STS_ERR;
143  }
144  }
145  CATCH_ANY {
146  result = STS_ERR;
147  }
148  FINALLY {
149  bn_free(m);
150  bn_free(t);
151  }
152 
153  return result;
154 }

156 int cp_rabin_dec(unsigned char *out, int *out_len, unsigned char *in,
157  int in_len, rabin_t prv) {
158  bn_t m, m0, m1, t, n;
159  int size, result = STS_OK;
160  unsigned char pad;
161 
162  if (in_len < 0 || in_len < RABIN_PAD_LEN) {
163  return STS_ERR;
164  }
165 
166  bn_null(m);
167  bn_null(m0);
168  bn_null(m1);
169  bn_null(n);
170  bn_null(t);
171 
172  TRY {
173  bn_new(m);
174  bn_new(m0);
175  bn_new(m1);
176  bn_new(n);
177  bn_new(t);
178 
179  bn_read_bin(m, in, in_len);
180 
181  bn_add_dig(t, prv->p, 1);
182  bn_rsh(t, t, 2);
183  bn_mxp(m0, m, t, prv->p);
184 
185  bn_add_dig(t, prv->q, 1);
186  bn_rsh(t, t, 2);
187  bn_mxp(m1, m, t, prv->q);
188 
189  bn_mul(m, prv->dp, prv->p);
190  bn_mul(m, m, m1);
191  bn_mul(t, prv->dq, prv->q);
192  bn_mul(t, t, m0);
193  bn_add(m0, m, t);
194  bn_mod(m0, m0, prv->n);
195  if (bn_sign(m0) == BN_NEG) {
196  bn_add(m0, m0, prv->n);
197  }
198  bn_sub(m1, m, t);
199  bn_mod(m1, m1, prv->n);
200  if (bn_sign(m1) == BN_NEG) {
201  bn_add(m1, m1, prv->n);
202  }
203 
204  bn_mod_2b(m, m0, 8 * RABIN_PAD_LEN);
205  bn_rsh(t, m0, 8 * RABIN_PAD_LEN);
206  bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
207  if (bn_cmp(t, m) == CMP_EQ) {
208  bn_rsh(m, m0, 8 * RABIN_PAD_LEN);
209  } else {
210  bn_sub(m0, prv->n, m0);
211  bn_mod_2b(m, m0, 8 * RABIN_PAD_LEN);
212  bn_rsh(t, m0, 8 * RABIN_PAD_LEN);
213  bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
214  if (bn_cmp(t, m) == CMP_EQ) {
215  bn_rsh(m, m0, 8 * RABIN_PAD_LEN);
216  } else {
217  bn_mod_2b(m, m1, 8 * RABIN_PAD_LEN);
218  bn_rsh(t, m1, 8 * RABIN_PAD_LEN);
219  bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
220  if (bn_cmp(t, m) == CMP_EQ) {
221  bn_rsh(m, m1, 8 * RABIN_PAD_LEN);
222  } else {
223  bn_sub(m1, prv->n, m1);
224  bn_mod_2b(m, m1, 8 * RABIN_PAD_LEN);
225  bn_rsh(t, m1, 8 * RABIN_PAD_LEN);
226  bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
227  if (bn_cmp(t, m) == CMP_EQ) {
228  bn_rsh(m, m1, 8 * RABIN_PAD_LEN);
229  } else {
230  result = STS_ERR;
231  }
232  }
233  }
234  }
235 
236  if (result == STS_OK) {
237  bn_size_bin(&size, prv->n);
238  size--;
239  bn_rsh(t, m, 8 * size);
240 
241  if (!bn_is_zero(t)) {
242  result = STS_ERR;
243  } else {
244  do {
245  size--;
246  bn_rsh(t, m, 8 * size);
247  pad = (unsigned char)t->dp[0];
248  } while (pad == 0);
249 
250  if (pad != RABIN_PAD) {
251  result = STS_ERR;
252  } else {
253  bn_mod_2b(m, m, size * 8);
254  }
255  }
256 
257  if (size <= *out_len) {
258  *out_len = size;
259  memset(out, 0, size);
260  bn_write_bin(out, size, m);
261  } else {
262  result = STS_ERR;
263  }
264  }
265  }
266  CATCH_ANY {
267  result = STS_ERR;
268  }
269  FINALLY {
270  bn_free(m);
271  bn_free(m0);
272  bn_free(m1);
273  bn_free(n);
274  bn_free(t);
275  }
276 
277  return result;
278 }
