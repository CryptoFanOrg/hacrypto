48 static void ep_dbl_basic_imp(ep_t r, fp_t s, const ep_t p) {
49  fp_t t0, t1, t2;
50 
51  fp_null(t0);
52  fp_null(t1);
53  fp_null(t2);
54 
55  TRY {
56  fp_new(t0);
57  fp_new(t1);
58  fp_new(t2);
59 
60  /* t0 = 1/2 * y1. */
61  fp_dbl(t0, p->y);
62  fp_inv(t0, t0);
63 
64  /* t1 = 3 * x1^2 + a. */
65  fp_sqr(t1, p->x);
66  fp_copy(t2, t1);
67  fp_dbl(t1, t1);
68  fp_add(t1, t1, t2);
69 
70  switch (ep_curve_opt_a()) {
71  case OPT_ZERO:
72  break;
73  case OPT_ONE:
74  fp_add_dig(t1, t1, (dig_t)1);
75  break;
76 #if FP_RDC != MONTY
77  case OPT_DIGIT:
78  fp_add_dig(t1, t1, ep_curve_get_a()[0]);
79  break;
80 #endif
81  default:
82  fp_add(t1, t1, ep_curve_get_a());
83  break;
84  }
85 
86  /* t1 = (3 * x1^2 + a)/(2 * y1). */
87  fp_mul(t1, t1, t0);
88 
89  if (s != NULL) {
90  fp_copy(s, t1);
91  }
92 
93  /* t2 = t1^2. */
94  fp_sqr(t2, t1);
95 
96  /* x3 = t1^2 - 2 * x1. */
97  fp_dbl(t0, p->x);
98  fp_sub(t0, t2, t0);
99 
100  /* y3 = t1 * (x1 - x3) - y1. */
101  fp_sub(t2, p->x, t0);
102  fp_mul(t1, t1, t2);
103  fp_sub(r->y, t1, p->y);
104 
105  fp_copy(r->x, t0);
106  fp_copy(r->z, p->z);
107 
108  r->norm = 1;
109  }
110  CATCH_ANY {
111  THROW(ERR_CAUGHT);
112  }
113  FINALLY {
114  fp_free(t0);
115  fp_free(t1);
116  fp_free(t2);
117  }
118 }

131 static void ep_dbl_projc_imp(ep_t r, const ep_t p) {
132  fp_t t0, t1, t2, t3, t4, t5;
133 
134  fp_null(t1);
135  fp_null(t2);
136  fp_null(t3);
137  fp_null(t4);
138  fp_null(t5);
139 
140  TRY {
141 
142  fp_new(t0);
143  fp_new(t1);
144  fp_new(t2);
145  fp_new(t3);
146  fp_new(t4);
147  fp_new(t5);
148 
149  if (!p->norm && ep_curve_opt_a() == OPT_MINUS3) {
150  /* dbl-2001-b formulas: 3M + 5S + 8add + 1*4 + 2*8 + 1*3 */
151  /* http://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-3.html#doubling-dbl-2001-b */
152 
153  /* t0 = delta = z1^2. */
154  fp_sqr(t0, p->z);
155 
156  /* t1 = gamma = y1^2. */
157  fp_sqr(t1, p->y);
158 
159  /* t2 = beta = x1 * y1^2. */
160  fp_mul(t2, p->x, t1);
161 
162  /* t3 = alpha = 3 * (x1 - z1^2) * (x1 + z1^2). */
163  fp_sub(t3, p->x, t0);
164  fp_add(t4, p->x, t0);
165  fp_mul(t4, t3, t4);
166  fp_dbl(t3, t4);
167  fp_add(t3, t3, t4);
168 
169  /* x3 = alpha^2 - 8 * beta. */
170  fp_dbl(t2, t2);
171  fp_dbl(t2, t2);
172  fp_dbl(t5, t2);
173  fp_sqr(r->x, t3);
174  fp_sub(r->x, r->x, t5);
175 
176  /* z3 = (y1 + z1)^2 - gamma - delta. */
177  fp_add(r->z, p->y, p->z);
178  fp_sqr(r->z, r->z);
179  fp_sub(r->z, r->z, t1);
180  fp_sub(r->z, r->z, t0);
181 
182  /* y3 = alpha * (4 * beta - x3) - 8 * gamma^2. */
183  fp_dbl(t1, t1);
184  fp_sqr(t1, t1);
185  fp_dbl(t1, t1);
186  fp_sub(r->y, t2, r->x);
187  fp_mul(r->y, r->y, t3);
188  fp_sub(r->y, r->y, t1);
189  } else if (ep_curve_opt_a() == OPT_ZERO) {
190  /* dbl-2009-l formulas: 2M + 5S + 6add + 1*8 + 3*2 + 1*3. */
191  /* http://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-0.html#doubling-dbl-2009-l */
192 
193  /* A = X1^2 */
194  fp_sqr(t0, p->x);
195 
196  /* B = Y1^2 */
197  fp_sqr(t1, p->y);
198 
199  /* C = B^2 */
200  fp_sqr(t2, t1);
201 
202  /* D = 2*((X1+B)^2-A-C) */
203  fp_add(t1, t1, p->x);
204  fp_sqr(t1, t1);
205  fp_sub(t1, t1, t0);
206  fp_sub(t1, t1, t2);
207  fp_dbl(t1, t1);
208 
209  /* E = 3*A */
210  fp_dbl(t3, t0);
211  fp_add(t0, t3, t0);
212 
213  /* F = E^2 */
214  fp_sqr(t3, t0);
215 
216  /* Z3 = 2*Y1*Z1 */
217  fp_mul(r->z, p->y, p->z);
218  fp_dbl(r->z, r->z);
219 
220  /* X3 = F-2*D */
221  fp_sub(r->x, t3, t1);
222  fp_sub(r->x, r->x, t1);
223 
224  /* Y3 = E*(D-X3)-8*C */
225  fp_sub(r->y, t1, r->x);
226  fp_mul(r->y, r->y, t0);
227  fp_dbl(t2, t2);
228  fp_dbl(t2, t2);
229  fp_dbl(t2, t2);
230  fp_sub(r->y, r->y, t2);
231  } else {
232  /* dbl-2007-bl formulas: 1M + 8S + 1*a + 10add + 1*8 + 2*2 + 1*3 */
233  /* http://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian.html#doubling-dbl-2007-bl */
234 
235  /* t0 = x1^2, t1 = y1^2, t2 = y1^4. */
236  fp_sqr(t0, p->x);
237  fp_sqr(t1, p->y);
238  fp_sqr(t2, t1);
239 
240  if (!p->norm) {
241  /* t3 = z1^2. */
242  fp_sqr(t3, p->z);
243 
244  if (ep_curve_get_a() == OPT_ZERO) {
245  /* z3 = 2 * y1 * z1. */
246  fp_mul(r->z, p->y, p->z);
247  fp_dbl(r->z, r->z);
248  } else {
249  /* z3 = (y1 + z1)^2 - y1^2 - z1^2. */
250  fp_add(r->z, p->y, p->z);
251  fp_sqr(r->z, r->z);
252  fp_sub(r->z, r->z, t1);
253  fp_sub(r->z, r->z, t3);
254  }
255  } else {
256  /* z3 = 2 * y1. */
257  fp_dbl(r->z, p->y);
258  }
259 
260  /* t4 = S = 2*((x1 + y1^2)^2 - x1^2 - y1^4). */
261  fp_add(t4, p->x, t1);
262  fp_sqr(t4, t4);
263  fp_sub(t4, t4, t0);
264  fp_sub(t4, t4, t2);
265  fp_dbl(t4, t4);
266 
267  /* t5 = M = 3 * x1^2 + a * z1^4. */
268  fp_dbl(t5, t0);
269  fp_add(t5, t5, t0);
270  if (!p->norm) {
271  fp_sqr(t3, t3);
272  switch (ep_curve_opt_a()) {
273  case OPT_ZERO:
274  break;
275  case OPT_ONE:
276  fp_add(t5, t5, ep_curve_get_a());
277  break;
278  case OPT_DIGIT:
279  fp_mul_dig(t1, t3, ep_curve_get_a()[0]);
280  fp_add(t5, t5, t1);
281  break;
282  default:
283  fp_mul(t1, ep_curve_get_a(), t3);
284  fp_add(t5, t5, t1);
285  break;
286  }
287  } else {
288  switch (ep_curve_opt_a()) {
289  case OPT_ZERO:
290  break;
291  case OPT_ONE:
292  fp_add_dig(t5, t5, (dig_t)1);
293  break;
294  case OPT_DIGIT:
295  fp_add_dig(t5, t5, ep_curve_get_a()[0]);
296  break;
297  default:
298  fp_add(t5, t5, ep_curve_get_a());
299  break;
300  }
301  }
302 
303  /* x3 = T = M^2 - 2 * S. */
304  fp_sqr(r->x, t5);
305  fp_dbl(t1, t4);
306  fp_sub(r->x, r->x, t1);
307 
308  /* y3 = M * (S - T) - 8 * y1^4. */
309  fp_dbl(t2, t2);
310  fp_dbl(t2, t2);
311  fp_dbl(t2, t2);
312  fp_sub(t4, t4, r->x);
313  fp_mul(t5, t5, t4);
314  fp_sub(r->y, t5, t2);
315  }
316 
317  r->norm = 0;
318  }
319  CATCH_ANY {
320  THROW(ERR_CAUGHT);
321  }
322  FINALLY {
323  fp_free(t0);
324  fp_free(t1);
325  fp_free(t2);
326  fp_free(t3);
327  fp_free(t4);
328  fp_free(t5);
329  }
330 }
