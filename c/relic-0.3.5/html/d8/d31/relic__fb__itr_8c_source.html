59 void fb_itr_pre_quick(fb_t *t, int b) {
60  int i, j, k;
61  fb_t r;
62 
63  fb_null(r);
64 
65  TRY {
66  fb_new(r);
67 
68  for (i = 0; i < FB_DIGS * FB_DIGIT; i += 4) {
69  for (j = 0; j < 16; j++) {
70  fb_zero(r);
71  fb_set_dig(r, j);
72  fb_lsh(r, r, i);
73  if (b >= 0) {
74  for (k = 0; k < b; k++) {
75  fb_sqr(r, r);
76  }
77  } else {
78  for (k = 0; k < -b; k++) {
79  fb_srt(r, r);
80  }
81  }
82 
83 #if ALLOC == AUTO
84  fb_copy((dig_t *)t + (4 * i + j) * FB_DIGS, r);
85 #else
86  fb_copy(t[4 * i + j], r);
87 #endif
88  }
89  }
90  } CATCH_ANY {
91  THROW(ERR_CAUGHT);
92  } FINALLY {
93  fb_free(r);
94  }
