41 void md_map_sh224(unsigned char *hash, const unsigned char *msg, int len) {
42  SHA224Context ctx;
43 
44  if (SHA224Reset(&ctx) != shaSuccess) {
45  THROW(ERR_NO_VALID);
46  }
47  if (SHA224Input(&ctx, msg, len) != shaSuccess) {
48  THROW(ERR_NO_VALID);
49  }
50  if (SHA224Result(&ctx, hash) != shaSuccess) {
51  THROW(ERR_NO_VALID);
52  }
53 }
