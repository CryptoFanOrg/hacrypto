45 static void fp_prime_set(const bn_t p) {
46  dv_t s, q;
47  bn_t t;
48  ctx_t *ctx = core_get();
49 
50  if (p->used != FP_DIGS) {
51  THROW(ERR_NO_VALID);
52  }
53 
54  dv_null(s);
55  bn_null(t);
56  dv_null(q);
57 
58  TRY {
59  dv_new(s);
60  bn_new(t);
61  dv_new(q);
62 
63  bn_copy(&(ctx->prime), p);
64 
65  bn_mod_dig(&(ctx->mod8), &(ctx->prime), 8);
66 
67  switch (ctx->mod8) {
68  case 3:
69  case 7:
70  ctx->qnr = -1;
71  /* The current code for extensions of Fp^3 relies on prime_qnr
72  * being also a cubic non-residue. */
73  ctx->cnr = 0;
74  break;
75  case 1:
76  case 5:
77  ctx->qnr = ctx->cnr = -2;
78  break;
79  default:
80  ctx->qnr = ctx->cnr = 0;
81  THROW(ERR_NO_VALID);
82  break;
83  }
84 #ifdef FP_QNRES
85  if (ctx->mod8 != 3) {
86  THROW(ERR_NO_VALID);
87  }
88 #endif
89 
90 #if FP_RDC == MONTY || !defined(STRIP)
91  bn_mod_pre_monty(t, &(ctx->prime));
92  ctx->u = t->dp[0];
93  dv_zero(s, 2 * FP_DIGS);
94  s[2 * FP_DIGS] = 1;
95  dv_zero(q, 2 * FP_DIGS + 1);
96  dv_copy(q, ctx->prime.dp, FP_DIGS);
97  bn_divn_low(t->dp, ctx->conv.dp, s, 2 * FP_DIGS + 1, q, FP_DIGS);
98  ctx->conv.used = FP_DIGS;
99  bn_trim(&(ctx->conv));
100  bn_set_dig(&(ctx->one), 1);
101  bn_lsh(&(ctx->one), &(ctx->one), ctx->prime.used * BN_DIGIT);
102  bn_mod(&(ctx->one), &(ctx->one), &(ctx->prime));
103 #endif
104  fp_prime_calc();
105  }
106  CATCH_ANY {
107  THROW(ERR_CAUGHT);
108  }
109  FINALLY {
110  bn_free(t);
111  dv_free(s);
112  dv_free(q);
113  }
114 }

121 static void fp2_calc() {
122  bn_t e;
123  fp2_t t0;
124  fp2_t t1;
125  ctx_t *ctx = core_get();
126 
127  bn_null(e);
128  fp2_null(t0);
129  fp2_null(t1);
130 
131  TRY {
132  bn_new(e);
133  fp2_new(t0);
134  fp2_new(t1);
135 
136  fp2_zero(t0);
137  fp_set_dig(t0[0], 1);
138  fp2_mul_nor(t0, t0);
139  e->used = FP_DIGS;
140  dv_copy(e->dp, fp_prime_get(), FP_DIGS);
141  bn_sub_dig(e, e, 1);
142  bn_div_dig(e, e, 6);
143  fp2_exp(t0, t0, e);
144 #if ALLOC == AUTO
145  fp2_copy(ctx->fp2_p[0], t0);
146  fp2_sqr(ctx->fp2_p[1], ctx->fp2_p[0]);
147  fp2_mul(ctx->fp2_p[2], ctx->fp2_p[1], ctx->fp2_p[0]);
148  fp2_sqr(ctx->fp2_p[3], ctx->fp2_p[1]);
149  fp2_mul(ctx->fp2_p[4], ctx->fp2_p[3], ctx->fp2_p[0]);
150 #else
151  fp_copy(ctx->fp2_p[0][0], t0[0]);
152  fp_copy(ctx->fp2_p[0][1], t0[1]);
153  fp2_sqr(t1, t0);
154  fp_copy(ctx->fp2_p[1][0], t1[0]);
155  fp_copy(ctx->fp2_p[1][1], t1[1]);
156  fp2_mul(t1, t1, t0);
157  fp_copy(ctx->fp2_p[2][0], t1[0]);
158  fp_copy(ctx->fp2_p[2][1], t1[1]);
159  fp2_sqr(t1, t0);
160  fp2_sqr(t1, t1);
161  fp_copy(ctx->fp2_p[3][0], t1[0]);
162  fp_copy(ctx->fp2_p[3][1], t1[1]);
163  fp2_mul(t1, t1, t0);
164  fp_copy(ctx->fp2_p[4][0], t1[0]);
165  fp_copy(ctx->fp2_p[4][1], t1[1]);
166 #endif
167  fp2_frb(t1, t0, 1);
168  fp2_mul(t0, t1, t0);
169  fp_copy(ctx->fp2_p2[0], t0[0]);
170  fp_sqr(ctx->fp2_p2[1], ctx->fp2_p2[0]);
171  fp_mul(ctx->fp2_p2[2], ctx->fp2_p2[1], ctx->fp2_p2[0]);
172  fp_sqr(ctx->fp2_p2[3], ctx->fp2_p2[1]);
173 
174  for (int i = 0; i < 5; i++) {
175  fp_mul(ctx->fp2_p3[i][0], ctx->fp2_p2[i % 3], ctx->fp2_p[i][0]);
176  fp_mul(ctx->fp2_p3[i][1], ctx->fp2_p2[i % 3], ctx->fp2_p[i][1]);
177  }
178  } CATCH_ANY {
179  THROW(ERR_CAUGHT);
180  } FINALLY {
181  bn_free(e);
182  fp2_free(t0);
183  fp2_free(t1);
184  }
185 }

190 static void fp3_calc() {
191  bn_t e;
192  fp3_t t0, t1, t2;
193  ctx_t *ctx = core_get();
194 
195  bn_null(e);
196  fp3_null(t0);
197  fp3_null(t1);
198  fp3_null(t2);
199 
200  TRY {
201  bn_new(e);
202  fp3_new(t0);
203  fp3_new(t1);
204  fp3_new(t2);
205 
206  fp_set_dig(ctx->fp3_base[0], -fp_prime_get_cnr());
207  fp_neg(ctx->fp3_base[0], ctx->fp3_base[0]);
208  e->used = FP_DIGS;
209  dv_copy(e->dp, fp_prime_get(), FP_DIGS);
210  bn_sub_dig(e, e, 1);
211  bn_div_dig(e, e, 3);
212  fp_exp(ctx->fp3_base[0], ctx->fp3_base[0], e);
213  fp_sqr(ctx->fp3_base[1], ctx->fp3_base[0]);
214 
215  fp3_zero(t0);
216  fp_set_dig(t0[1], 1);
217  dv_copy(e->dp, fp_prime_get(), FP_DIGS);
218  bn_sub_dig(e, e, 1);
219  bn_div_dig(e, e, 6);
220 
221  /* t0 = u^((p-1)/6). */
222  fp3_exp(t0, t0, e);
223  fp_copy(ctx->fp3_p[0], t0[2]);
224  fp3_sqr(t1, t0);
225  fp_copy(ctx->fp3_p[1], t1[1]);
226  fp3_mul(t2, t1, t0);
227  fp_copy(ctx->fp3_p[2], t2[0]);
228  fp3_sqr(t2, t1);
229  fp_copy(ctx->fp3_p[3], t2[2]);
230  fp3_mul(t2, t2, t0);
231  fp_copy(ctx->fp3_p[4], t2[1]);
232 
233  fp_mul(ctx->fp3_p2[0], ctx->fp3_p[0], ctx->fp3_base[1]);
234  fp_mul(t0[0], ctx->fp3_p2[0], ctx->fp3_p[0]);
235  fp_neg(ctx->fp3_p2[0], t0[0]);
236  for (int i = -1; i > fp_prime_get_cnr(); i--) {
237  fp_sub(ctx->fp3_p2[0], ctx->fp3_p2[0], t0[0]);
238  }
239  fp_mul(ctx->fp3_p2[1], ctx->fp3_p[1], ctx->fp3_base[0]);
240  fp_mul(ctx->fp3_p2[1], ctx->fp3_p2[1], ctx->fp3_p[1]);
241  fp_sqr(ctx->fp3_p2[2], ctx->fp3_p[2]);
242  fp_mul(ctx->fp3_p2[3], ctx->fp3_p[3], ctx->fp3_base[1]);
243  fp_mul(t0[0], ctx->fp3_p2[3], ctx->fp3_p[3]);
244  fp_neg(ctx->fp3_p2[3], t0[0]);
245  for (int i = -1; i > fp_prime_get_cnr(); i--) {
246  fp_sub(ctx->fp3_p2[3], ctx->fp3_p2[3], t0[0]);
247  }
248  fp_mul(ctx->fp3_p2[4], ctx->fp3_p[4], ctx->fp3_base[0]);
249  fp_mul(ctx->fp3_p2[4], ctx->fp3_p2[4], ctx->fp3_p[4]);
250 
251  fp_mul(ctx->fp3_p3[0], ctx->fp3_p[0], ctx->fp3_base[0]);
252  fp_mul(t0[0], ctx->fp3_p3[0], ctx->fp3_p2[0]);
253  fp_neg(ctx->fp3_p3[0], t0[0]);
254  for (int i = -1; i > fp_prime_get_cnr(); i--) {
255  fp_sub(ctx->fp3_p3[0], ctx->fp3_p3[0], t0[0]);
256  }
257  fp_mul(ctx->fp3_p3[1], ctx->fp3_p[1], ctx->fp3_base[1]);
258  fp_mul(t0[0], ctx->fp3_p3[1], ctx->fp3_p2[1]);
259  fp_neg(ctx->fp3_p3[1], t0[0]);
260  for (int i = -1; i > fp_prime_get_cnr(); i--) {
261  fp_sub(ctx->fp3_p3[1], ctx->fp3_p3[1], t0[0]);
262  }
263  fp_mul(ctx->fp3_p3[2], ctx->fp3_p[2], ctx->fp3_p2[2]);
264  fp_mul(ctx->fp3_p3[3], ctx->fp3_p[3], ctx->fp3_base[0]);
265  fp_mul(t0[0], ctx->fp3_p3[3], ctx->fp3_p2[3]);
266  fp_neg(ctx->fp3_p3[3], t0[0]);
267  for (int i = -1; i > fp_prime_get_cnr(); i--) {
268  fp_sub(ctx->fp3_p3[3], ctx->fp3_p3[3], t0[0]);
269  }
270  fp_mul(ctx->fp3_p3[4], ctx->fp3_p[4], ctx->fp3_base[1]);
271  fp_mul(t0[0], ctx->fp3_p3[4], ctx->fp3_p2[4]);
272  fp_neg(ctx->fp3_p3[4], t0[0]);
273  for (int i = -1; i > fp_prime_get_cnr(); i--) {
274  fp_sub(ctx->fp3_p3[4], ctx->fp3_p3[4], t0[0]);
275  }
276  for (int i = 0; i < 5; i++) {
277  fp_mul(ctx->fp3_p4[i], ctx->fp3_p[i], ctx->fp3_p3[i]);
278  fp_mul(ctx->fp3_p5[i], ctx->fp3_p2[i], ctx->fp3_p3[i]);
279  }
280  } CATCH_ANY {
281  THROW(ERR_CAUGHT);
282  } FINALLY {
283  bn_free(e);
284  fp3_free(t0);
285  fp3_free(t1);
286  fp3_free(t2);
287  }
288 }

370 void fp_prime_set_pmers(const int *f, int len) {
371  bn_t p, t;
372  ctx_t *ctx = core_get();
373 
374  bn_null(p);
375  bn_null(t);
376 
377  TRY {
378  bn_new(p);
379  bn_new(t);
380 
381  if (len >= MAX_TERMS) {
382  THROW(ERR_NO_VALID);
383  }
384 
385  for (int i = len - 1; i > 0; i--) {
386  if (f[i] > 0) {
387  bn_set_2b(t, f[i]);
388  bn_add(p, p, t);
389  } else {
390  bn_set_2b(t, -f[i]);
391  bn_sub(p, p, t);
392  }
393  }
394  if (f[0] > 0) {
395  bn_add_dig(p, p, f[0]);
396  } else {
397  bn_sub_dig(p, p, -f[0]);
398  }
399 
400  for (int i = 0; i < len; i++) {
401  ctx->sps[i] = f[i];
402  }
403  ctx->sps[len] = 0;
404  ctx->sps_len = len;
405  fp_prime_set(p);
406  }
407  CATCH_ANY {
408  THROW(ERR_CAUGHT);
409  }
410  FINALLY {
411  bn_free(p);
412  bn_free(t);
413  }
414 }

427 void fp_prime_conv(fp_t c, const bn_t a) {
428  bn_t t;
429 
430  bn_null(t);
431 
432  TRY {
433  bn_new(t);
434 
435 #if FP_RDC == MONTY
436  bn_mod(t, a, &(core_get()->prime));
437  bn_lsh(t, t, FP_DIGS * FP_DIGIT);
438  bn_mod(t, t, &(core_get()->prime));
439  dv_copy(c, t->dp, FP_DIGS);
440 #else
441  if (a->used > FP_DIGS) {
442  THROW(ERR_NO_PRECI);
443  }
444 
445  bn_mod(t, a, &(core_get()->prime));
446 
447  if (bn_is_zero(t)) {
448  fp_zero(c);
449  } else {
450  int i;
451  for (i = 0; i < t->used; i++) {
452  c[i] = t->dp[i];
453  }
454  for (; i < FP_DIGS; i++) {
455  c[i] = 0;
456  }
457  }
458  (void)t;
459 #endif
460  }
461  CATCH_ANY {
462  THROW(ERR_CAUGHT);
463  }
464  FINALLY {
465  bn_free(t);
466  }
