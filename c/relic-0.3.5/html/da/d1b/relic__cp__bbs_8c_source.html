40 void cp_bbs_gen(bn_t d, g2_t q, gt_t z) {
41  bn_t n;
42  g1_t g;
43 
44  bn_null(n);
45  g1_null(g);
46 
47  TRY {
48  bn_new(n);
49  g1_new(g);
50 
51  g1_get_gen(g);
52  g2_get_gen(q);
53 
54  /* z = e(g1, g2). */
55  pc_map(z, g, q);
56 
57  g2_get_ord(n);
58 
59  do {
60  bn_rand(d, BN_POS, 2 * pc_param_level());
61  bn_mod(d, d, n);
62  } while (bn_is_zero(d));
63 
64  /* q = d * g2. */
65  g2_mul_gen(q, d);
66  g2_norm(q, q);
67  }
68  CATCH_ANY {
69  THROW(ERR_CAUGHT);
70  }
71  FINALLY {
72  bn_free(n);
73  }
74 }

76 void cp_bbs_sig(g1_t s, unsigned char *msg, int len, int hash, bn_t d) {
77  bn_t m, n, r;
78  unsigned char h[MD_LEN];
79 
80  bn_null(m);
81  bn_null(n);
82  bn_null(r);
83 
84  TRY {
85  bn_new(m);
86  bn_new(n);
87  bn_new(r);
88 
89  g1_get_ord(n);
90 
91  /* m = H(msg). */
92  if (hash) {
93  bn_read_bin(m, msg, len);
94  } else {
95  md_map(h, msg, len);
96  bn_read_bin(m, h, MD_LEN);
97  }
98  bn_mod(m, m, n);
99 
100  /* m = 1/(m + d) mod n. */
101  bn_add(m, m, d);
102  bn_gcd_ext(r, m, NULL, m, n);
103  if (bn_sign(m) == BN_NEG) {
104  bn_add(m, m, n);
105  }
106  /* s = 1/(m+d) * g1. */
107  g1_mul_gen(s, m);
108  }
109  CATCH_ANY {
110  THROW(ERR_CAUGHT);
111  }
112  FINALLY {
113  bn_free(m);
114  bn_free(n);
115  bn_free(r);
116  }
117 }
