50 static void pp_mil_k2(fp2_t r, ep_t t, ep_t p, ep_t q, bn_t a) {
51  fp2_t l;
52  ep_t _q;
53 
54  fp2_null(l);
55  ep_null(_q);
56 
57  TRY {
58  fp2_new(l);
59  ep_new(_q);
60 
61  fp2_zero(l);
62  fp2_zero(r);
63  fp_set_dig(r[0], 1);
64  ep_copy(t, p);
65 
66  ep_neg(_q, q);
67 
68  for (int i = bn_bits(a) - 2; i >= 0; i--) {
69  fp2_sqr(r, r);
70  pp_dbl_k2(l, t, t, _q);
71  fp2_mul(r, r, l);
72  if (bn_test_bit(a, i)) {
73  pp_add_k2(l, t, p, q);
74  fp2_mul(r, r, l);
75  }
76  }
77  }
78  CATCH_ANY {
79  THROW(ERR_CAUGHT);
80  }
81  FINALLY {
82  fp2_free(l);
83  ep_free(_q);
84  }
85 }

436 void pp_map_oatep_k12(fp12_t r, ep_t p, ep2_t q) {
437  ep2_t t;
438  bn_t a;
439  int len = FP_BITS, s[FP_BITS];
440 
441  ep2_null(t);
442  bn_null(a);
443 
444  TRY {
445  ep2_new(t);
446  bn_new(a);
447 
448  fp_param_get_var(a);
449  fp_param_get_map(s, &len);
450 
451  switch (ep_param_get()) {
452  case BN_P158:
453  case BN_P254:
454  case BN_P256:
455  case BN_P638:
456  /* r = f_{|a|,Q}(P). */
457  pp_mil_sps_k12(r, t, q, p, s, len);
458  if (bn_sign(a) == BN_NEG) {
459  /* f_{-a,Q}(P) = 1/f_{a,Q}(P). */
460  fp12_inv_uni(r, r);
461  ep2_neg(t, t);
462  }
463  pp_fin_k12_oatep(r, t, q, p);
464  pp_exp_k12(r, r);
465  break;
466  case B12_P638:
467  /* r = f_{|a|,Q}(P). */
468  pp_mil_sps_k12(r, t, q, p, s, len);
469  if (bn_sign(a) == BN_NEG) {
470  fp12_inv_uni(r, r);
471  ep2_neg(t, t);
472  }
473  pp_exp_k12(r, r);
474  break;
475  }
476  }
477  CATCH_ANY {
478  THROW(ERR_CAUGHT);
479  }
480  FINALLY {
481  ep2_free(t);
482  bn_free(a);
483  }
484 }
