46 void fb_exp_basic(fb_t c, const fb_t a, const bn_t b) {
47  int i, l;
48  fb_t r;
49 
50  fb_null(r);
51 
52  TRY {
53  fb_new(r);
54 
55  l = bn_bits(b);
56 
57  fb_copy(r, a);
58 
59  for (i = l - 2; i >= 0; i--) {
60  fb_sqr(r, r);
61  if (bn_test_bit(b, i)) {
62  fb_mul(r, r, a);
63  }
64  }
65 
66  fb_copy(c, r);
67  }
68  CATCH_ANY {
69  THROW(ERR_CAUGHT);
70  }
71  FINALLY {
72  fb_free(r);
73  }
74 }

80 void fb_exp_slide(fb_t c, const fb_t a, const bn_t b) {
81  fb_t t[1 << (FB_WIDTH - 1)], r;
82  int i, j, l;
83  unsigned char win[FB_BITS + 1];
84 
85  fb_null(r);
86 
87  /* Initialize table. */
88  for (i = 0; i < (1 << (FB_WIDTH - 1)); i++) {
89  fb_null(t[i]);
90  }
91 
92  TRY {
93  for (i = 0; i < (1 << (FB_WIDTH - 1)); i++) {
94  fb_new(t[i]);
95  }
96  fb_new(r);
97 
98  fb_copy(t[0], a);
99  fb_sqr(r, a);
100 
101  /* Create table. */
102  for (i = 1; i < 1 << (FB_WIDTH - 1); i++) {
103  fb_mul(t[i], t[i - 1], r);
104  }
105 
106  fb_set_dig(r, 1);
107  l = FB_BITS + 1;
108  bn_rec_slw(win, &l, b, FB_WIDTH);
109  for (i = 0; i < l; i++) {
110  if (win[i] == 0) {
111  fb_sqr(r, r);
112  } else {
113  for (j = 0; j < util_bits_dig(win[i]); j++) {
114  fb_sqr(r, r);
115  }
116  fb_mul(r, r, t[win[i] >> 1]);
117  }
118  }
119 
120  fb_copy(c, r);
121  }
122  CATCH_ANY {
123  THROW(ERR_CAUGHT);
124  }
125  FINALLY {
126  for (i = 0; i < (1 << (FB_WIDTH - 1)); i++) {
127  fb_free(t[i]);
128  }
129  fb_free(r);
130  }
131 }

137 void fb_exp_monty(fb_t c, const fb_t a, const bn_t b) {

140  dig_t buf;
141  int bitcnt, digidx, j;
142 
143  fb_null(t[0]);
144  fb_null(t[1]);
145 
146  TRY {
147  fb_new(t0);
148  fb_new(t1);
149 
150  fb_set_dig(t0, 1);
151  fb_copy(t1, a);
152 
153  /* This trick avoid buggy compilers with alignment issues. */
154  t[0] = t0;
155  t[1] = t1;
156 
157  /* Set initial mode and bitcnt, */
158  bitcnt = 1;
159  buf = 0;
160  digidx = b->used - 1;
161 
162  for (;;) {
163  /* Grab next digit as required. */
164  if (--bitcnt == 0) {
165  /* If digidx == -1 we are out of digits so break. */
166  if (digidx == -1) {
167  break;
168  }
169  /* Read next digit and reset bitcnt. */
170  buf = b->dp[digidx--];
171  bitcnt = (int)FB_DIGIT;
172  }
173 
174  /* Grab the next msb from the exponent. */
175  j = (buf >> (FB_DIGIT - 1)) & 0x01;
176  buf <<= (dig_t)1;
177 
178  fb_mul(t[j ^ 1], t[0], t[1]);
179  fb_sqr(t[j], t[j]);
180  }
181 
182  fb_copy(c, t[0]);
183 
184  } CATCH_ANY {
185  THROW(ERR_CAUGHT);
186  }
187  FINALLY {
188  fb_free(t[1]);
189  fb_free(t[0]);
190  }
191 }
