48 static void bn_div_imp(bn_t c, bn_t d, const bn_t a, const bn_t b) {
49  bn_t q, x, y, r;
50  int sign;
51 
52  bn_null(q);
53  bn_null(x);
54  bn_null(y);
55  bn_null(r);
56 
57  /* If a < b, we're done. */
58  if (bn_cmp_abs(a, b) == CMP_LT) {
59  if (d != NULL) {
60  bn_copy(d, a);
61  }
62  if (c != NULL) {
63  bn_zero(c);
64  }
65  return;
66  }
67 
68  TRY {
69  bn_new(x);
70  bn_new(y);
71  bn_new_size(q, a->used + 1);
72  bn_new(r);
73  bn_zero(q);
74  bn_zero(r);
75  bn_abs(x, a);
76  bn_abs(y, b);
77 
78  /* Find the sign. */
79  sign = (a->sign == b->sign ? BN_POS : BN_NEG);
80 
81  bn_divn_low(q->dp, r->dp, x->dp, a->used, y->dp, b->used);
82 
83  /* We have the quotient in q and the remainder in r. */
84  if (c != NULL) {
85  q->used = a->used - b->used + 1;
86  q->sign = sign;
87  bn_trim(q);
88  bn_copy(c, q);
89  }
90 
91  if (d != NULL) {
92  r->used = b->used;
93  r->sign = a->sign;
94  bn_trim(r);
95  bn_copy(d, r);
96  }
97  }
98  CATCH_ANY {
99  THROW(ERR_CAUGHT);
100  }
101  FINALLY {
102  bn_free(r);
103  bn_free(q);
104  bn_free(x);
105  bn_free(y);
106  }
107 }

131  bn_null(q);
132 
133  if (b == 0) {
