96 void ep_rhs(fp_t rhs, const ep_t p) {
97  fp_t t0;
98  fp_t t1;
99 
100  fp_null(t0);
101  fp_null(t1);
102 
103  TRY {
104  fp_new(t0);
105  fp_new(t1);
106 
107  /* t0 = x1^2. */
108  fp_sqr(t0, p->x);
109  /* t1 = x1^3. */
110  fp_mul(t1, t0, p->x);
111 
112  /* t1 = x1^3 + a * x1 + b. */
113  switch (ep_curve_opt_a()) {
114  case OPT_ZERO:
115  break;
116  case OPT_ONE:
117  fp_add(t1, t1, p->x);
118  break;
119 #if FP_RDC != MONTY
120  case OPT_DIGIT:
121  fp_mul_dig(t0, p->x, ep_curve_get_a()[0]);
122  fp_add(t1, t1, t0);
123  break;
124 #endif
125  default:
126  fp_mul(t0, p->x, ep_curve_get_a());
127  fp_add(t1, t1, t0);
128  break;
129  }
130 
131  switch (ep_curve_opt_b()) {
132  case OPT_ZERO:
133  break;
134  case OPT_ONE:
135  fp_add_dig(t1, t1, 1);
136  break;
137 #if FP_RDC != MONTY
138  case OPT_DIGIT:
139  fp_add_dig(t1, t1, ep_curve_get_b()[0]);
140  break;
141 #endif
142  default:
143  fp_add(t1, t1, ep_curve_get_b());
144  break;
145  }
146 
147  fp_copy(rhs, t1);
148 
149  } CATCH_ANY {
150  THROW(ERR_CAUGHT);
151  } FINALLY {
152  fp_free(t0);
153  fp_free(t1);
154  }
155 }

157 void ep_tab(ep_t *t, const ep_t p, int w) {
158  if (w > 2) {
159  ep_dbl(t[0], p);
160 #if defined(EP_MIXED)
161  ep_norm(t[0], t[0]);
162 #endif
163  ep_add(t[1], t[0], p);
164  for (int i = 2; i < (1 << (w - 2)); i++) {
165  ep_add(t[i], t[i - 1], t[0]);
166  }
167 #if defined(EP_MIXED)
168  ep_norm_sim(t + 1, (const ep_t *)t + 1, (1 << (w - 2)) - 1);
169 #endif
170  }
171  ep_copy(t[0], p);
172 }

174 int ep_is_valid(const ep_t p) {
175  ep_t t;
176  int r = 0;
177 
178  ep_null(t);
179 
180  TRY {
181  ep_new(t);
182 
183  ep_norm(t, p);
184 
185  ep_rhs(t->x, t);
186  fp_sqr(t->y, t->y);
187  r = (fp_cmp(t->x, t->y) == CMP_EQ) || ep_is_infty(p);
188  } CATCH_ANY {
189  THROW(ERR_CAUGHT);
190  } FINALLY {
191  ep_free(t);
192  }
193  return r;
194 }
