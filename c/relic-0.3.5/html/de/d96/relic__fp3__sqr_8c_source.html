42 void fp3_sqr_basic(fp3_t c, fp3_t a) {
43  dv_t t0, t1, t2, t3, t4, t5;
44 
45  dv_null(t0);
46  dv_null(t1);
47  dv_null(t2);
48  dv_null(t3);
49  dv_null(t4);
50  dv_null(t5);
51 
52  TRY {
53  dv_new(t0);
54  dv_new(t1);
55  dv_new(t2);
56  dv_new(t3);
57  dv_new(t4);
58  dv_new(t5);
59 
60  /* t0 = a_0^2. */
61  fp_sqrn_low(t0, a[0]);
62 
63  /* t1 = 2 * a_1 * a_2. */
64  fp_dbl(t2, a[1]);
65  fp_muln_low(t1, t2, a[2]);
66 
67  /* t2 = a_2^2. */
68  fp_sqrn_low(t2, a[2]);
69 
70  /* t3 = (a_0 + a_2 + a_1)^2, t4 = (a_0 + a_2 - a_1)^2. */
71  fp_add(t3, a[0], a[2]);
72  fp_add(t4, t3, a[1]);
73  fp_sub(t5, t3, a[1]);
74  fp_sqrn_low(t3, t4);
75  fp_sqrn_low(t4, t5);
76 
77  /* t4 = (t4 + t3)/2. */
78  fp_addd_low(t4, t4, t3);
79  fp_hlvd_low(t4, t4);
80 
81  /* t3 = t3 - t4 - t1. */
82  fp_addc_low(t5, t1, t4);
83  fp_subc_low(t3, t3, t5);
84 
85  /* c_2 = t4 - t0 - t2. */
86  fp_addc_low(t5, t0, t2);
87  fp_subc_low(t4, t4, t5);
88  fp_rdc(c[2], t4);
89 
90  /* c_0 = t0 + t1 * B. */
91  fp_subc_low(t0, t0, t1);
92  for (int i = -1; i > fp_prime_get_cnr(); i--) {
93  fp_subc_low(t0, t0, t1);
94  }
95  fp_rdc(c[0], t0);
96 
97  /* c_1 = t3 + t2 * B. */
98  fp_subc_low(t3, t3, t2);
99  for (int i = -1; i > fp_prime_get_cnr(); i--) {
100  fp_subc_low(t3, t3, t2);
101  }
102  fp_rdc(c[1], t3);
103  }
104  CATCH_ANY {
105  THROW(ERR_CAUGHT);
106  }
107  FINALLY {
108  dv_free(t0);
109  dv_free(t1);
110  dv_free(t2);
111  dv_free(t3);
112  dv_free(t4);
113  dv_free(t5);
114  }
115 }
