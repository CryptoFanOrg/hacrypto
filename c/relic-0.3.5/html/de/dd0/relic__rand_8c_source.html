88 void rand_init() {
89  unsigned char buf[SEED_SIZE];
90 
91  memset(core_get()->rand, 0, RAND_SIZE);
92 
93 #if SEED == ZERO
94 
95  memset(buf, 0, SEED_SIZE);
96 
97 #elif SEED == DEV || SEED == UDEV
98  int rand_fd, c, l;
99 
100  rand_fd = open(RAND_PATH, O_RDONLY);
101  if (rand_fd == -1) {
102  THROW(ERR_NO_FILE);
103  }
104 
105  l = 0;
106  do {
107  c = read(rand_fd, buf + l, SEED_SIZE - l);
108  l += c;
109  if (c == -1) {
110  THROW(ERR_NO_READ);
111  }
112  } while (l < SEED_SIZE);
113 
114  if (rand_fd != -1) {
115  close(rand_fd);
116  }
117 
118 #elif SEED == LIBC
119 
120 #if OPSYS == FREEBSD
121  srandom(1);
122  for (int i = 0; i < SEED_SIZE; i++) {
123  buf[i] = (unsigned char)random();
124  }
125 #else
126  srand(1);
127  for (int i = 0; i < SEED_SIZE; i++) {
128  buf[i] = (unsigned char)rand();
129  }
130 #endif
131 
132 #elif SEED == WCGR
133  HCRYPTPROV hCryptProv;
134 
135  if (!CryptAcquireContext(&hCryptProv, NULL, NULL, PROV_RSA_FULL, CRYPT_VERIFYCONTEXT | CRYPT_SILENT)) {
136  THROW(ERR_NO_FILE);
137  }
138  if (hCryptProv && !CryptGenRandom(hCryptProv, SEED_SIZE, buf)) {
139  THROW(ERR_NO_READ);
140  }
141  if (hCryptProv && !CryptReleaseContext(hCryptProv, 0)) {
142  THROW(ERR_NO_READ);
143  }
144 #endif
145 
146  rand_seed(buf, SEED_SIZE);
147 }

153 void rand_seed(unsigned char *buf, int size) {
154  int i;
155  ctx_t *ctx = core_get();
156 
157  if (size < SEED_SIZE) {
158  THROW(ERR_NO_VALID);
159  }
160 
161  /* Zero the current state. */
162  memset(ctx->rand, 0, sizeof(ctx->rand));
163 
164  /* XKEY = SEED */
165  for (i = 0; i < MIN(size, SEED_SIZE); i++) {
166  ctx->rand[i] = buf[i];
167  }
168 }

170 void rand_bytes(unsigned char *buf, int size) {
171  unsigned char carry, c0, c1, r0, r1;
172  int i, j;
173  unsigned char hash[MD_LEN_SHONE];
174  ctx_t *ctx = core_get();
175 
176  j = 0;
177  while (j < size) {
178  /* x = G(t, XKEY) */
179  md_map_shone_mid(ctx->rand, RAND_SIZE, hash);
180 
181  /* XKEY = (XKEY + x + 1) mod 2^b */
182  carry = 1;
183  for (i = SEED_SIZE - 1; i >= 0; i--) {
184  r0 = (unsigned char)(ctx->rand[i] + hash[i]);
185  c0 = (unsigned char)(r0 < hash[i] ? 1 : 0);
186  r1 = (unsigned char)(r0 + carry);
187  c1 = (unsigned char)(r1 < r0 ? 1 : 0);
188  carry = (unsigned char)(c0 | c1);
189  ctx->rand[i] = r1;
190  }
191  for (i = 0; i < SEED_SIZE && j < size; i++) {
192  buf[j] = hash[i];
193  j++;
194  }
195  }
196 }
