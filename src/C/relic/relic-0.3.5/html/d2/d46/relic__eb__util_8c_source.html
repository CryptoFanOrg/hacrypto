40 static void eb_mul_frb(eb_t r, const eb_t p, const signed char *k, int l) {
41  int i;
42  eb_t t;
43 
44  eb_null(t);
45 
46  TRY {
47  eb_new(t);
48 
49  if (k[l - 1] >= 0) {
50  eb_copy(t, p);
51  } else {
52  eb_neg(t, p);
53  }
54  for (i = l - 2; i >= 0; i--) {
55  eb_frb(t, t);
56  if (k[i] > 0) {
57  eb_add(t, t, p);
58  }
59  if (k[i] < 0) {
60  eb_sub(t, t, p);
61  }
62  }
63 
64  eb_norm(r, t);
65  }
66  CATCH_ANY {
67  THROW(ERR_CAUGHT);
68  }
69  FINALLY {
70  eb_free(t);
71  }
72 }

136 void eb_rhs(fb_t rhs, const eb_t p) {
137  fb_t t0, t1;
138 
139  fb_null(t0);
140  fb_null(t1);
141 
142  TRY {
143  fb_new(t0);
144  fb_new(t1);
145 
146  /* t0 = x1^2. */
147  fb_sqr(t0, p->x);
148  /* t1 = x1^3. */
149  fb_mul(t1, t0, p->x);
150 
151  if (eb_curve_is_super()) {
152  /* t1 = x1^3 + a * x1 + b. */
153  switch (eb_curve_opt_a()) {
154  case OPT_ZERO:
155  break;
156  case OPT_ONE:
157  fb_add(t1, t1, p->x);
158  break;
159  case OPT_DIGIT:
160  fb_mul_dig(t0, p->x, eb_curve_get_a()[0]);
161  fb_add(t1, t1, t0);
162  break;
163  default:
164  fb_mul(t0, p->x, eb_curve_get_a());
165  fb_add(t1, t1, t0);
166  break;
167  }
168  } else {
169  /* t1 = x1^3 + a * x1^2 + b. */
170  switch (eb_curve_opt_a()) {
171  case OPT_ZERO:
172  break;
173  case OPT_ONE:
174  fb_add(t1, t1, t0);
175  break;
176  case OPT_DIGIT:
177  fb_mul_dig(t0, t0, eb_curve_get_a()[0]);
178  fb_add(t1, t1, t0);
179  break;
180  default:
181  fb_mul(t0, t0, eb_curve_get_a());
182  fb_add(t1, t1, t0);
183  break;
184  }
185  }
186 
187  switch (eb_curve_opt_b()) {
188  case OPT_ZERO:
189  break;
190  case OPT_ONE:
191  fb_add_dig(t1, t1, 1);
192  break;
193  case OPT_DIGIT:
194  fb_add_dig(t1, t1, eb_curve_get_b()[0]);
195  break;
196  default:
197  fb_add(t1, t1, eb_curve_get_b());
198  break;
199  }
200 
201  fb_copy(rhs, t1);
202  }
203  CATCH_ANY {
204  THROW(ERR_CAUGHT);
205  }
206  FINALLY {
207  fb_free(t0);
208  fb_free(t1);
209  }
