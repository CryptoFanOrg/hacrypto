46 void ep2_mul_cof_bn(ep2_t r, ep2_t p) {
47  bn_t x;
48  ep2_t t0, t1, t2;
49 
50  ep2_null(t0);
51  ep2_null(t1);
52  ep2_null(t2);
53  bn_null(x);
54 
55  TRY {
56  ep2_new(t0);
57  ep2_new(t1);
58  ep2_new(t2);
59  bn_new(x);
60 
61  fp_param_get_var(x);
62 
63  /* Compute t0 = xP. */
64  ep2_mul(t0, p, x);
65  if (bn_sign(x) == BN_NEG) {
66  ep2_neg(t0, t0);
67  }
68 
69  /* Compute t1 = \psi(3xP). */
70  ep2_dbl(t1, t0);
71  ep2_add(t1, t1, t0);
72  ep2_norm(t1, t1);
73  ep2_frb(t1, t1, 1);
74 
75  /* Compute t2 = \psi^3(P) + t0 + t1 + \psi^2(xP). */
76  ep2_frb(t2, p, 2);
77  ep2_frb(t2, t2, 1);
78  ep2_add(t2, t2, t0);
79  ep2_add(t2, t2, t1);
80  ep2_frb(t1, t0, 2);
81  ep2_add(t2, t2, t1);
82 
83  ep2_norm(r, t2);
84  }
85  CATCH_ANY {
86  THROW(ERR_CAUGHT);
87  }
88  FINALLY {
89  ep2_free(t0);
90  ep2_free(t1);
91  ep2_free(t2);
92  bn_free(x);
93  }
94 }

104  ep2_t t0, t1, t2, t3;

106  ep2_null(t0);

113  ep2_new(t0);
114  ep2_new(t1);
115  ep2_new(t2);
116  ep2_new(t3);
117  bn_new(x);
118 
119  fp_param_get_var(x);
120 
121  /* Compute t0 = xP. */
122  ep2_mul(t0, p, x);
123  if (bn_sign(x) == BN_NEG) {
124  ep2_neg(t0, t0);
125  }
126  /* Compute t1 = [x^2]P. */
127  ep2_mul(t1, t0, x);
128  if (bn_sign(x) == BN_NEG) {
129  ep2_neg(t1, t1);
130  }
131 
132  /* t2 = (x^2 - x - 1)P = x^2P - x*P - P. */
133  ep2_sub(t2, t1, t0);
134  ep2_sub(t2, t2, p);
135  /* t3 = \psi(x - 1)P. */
136  ep2_sub(t3, t0, p);
137  ep2_norm(t3, t3);
138  ep2_frb(t3, t3, 1);
139  ep2_add(t2, t2, t3);
140  /* t3 = \psi^2(2P). */
141  ep2_dbl(t3, p);
142  ep2_norm(t3, t3);
143  ep2_frb(t3, t3, 2);
144  ep2_add(t2, t2, t3);
145  ep2_norm(r, t2);
146  }
147  CATCH_ANY {
148  THROW(ERR_CAUGHT);
149  }
150  FINALLY {
151  ep2_free(t0);
152  ep2_free(t1);
153  ep2_free(t2);
154  ep2_free(t3);

163 void ep2_map(ep2_t p, unsigned char *msg, int len) {
164  bn_t x;
165  fp2_t t0;
166  unsigned char digest[MD_LEN];
167 
168  bn_null(x);
169  fp2_null(t0);
170 
171  TRY {
172  bn_new(x);
173  fp2_new(t0);
174 
175  md_map(digest, msg, len);
176  bn_read_bin(x, digest, MIN(FP_BYTES, MD_LEN));
177 
178  fp_prime_conv(p->x[0], x);
179  fp_zero(p->x[1]);
180  fp_set_dig(p->z[0], 1);
181  fp_zero(p->z[1]);
182 
183  while (1) {
184  ep2_rhs(t0, p);
185 
186  if (fp2_srt(p->y, t0)) {
187  p->norm = 1;
188  break;
189  }
190 
191  fp_add_dig(p->x[0], p->x[0], 1);
192  }
193 
194  switch (ep_param_get()) {
195  case BN_P158:
196  case BN_P254:
197  case BN_P256:
198  case BN_P638:
199  ep2_mul_cof_bn(p, p);
200  break;
201  case B12_P638:
202  ep2_mul_cof_b12(p, p);
203  break;
204  }
205  }
206  CATCH_ANY {
207  THROW(ERR_CAUGHT);
208  }
209  FINALLY {
210  bn_free(x);
211  fp2_free(t0);
212  }
213 }
