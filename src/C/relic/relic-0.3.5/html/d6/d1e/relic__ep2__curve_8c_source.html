153 void ep2_curve_init(void) {
154  ctx_t *ctx = core_get();
155 
156 #ifdef EP_PRECO
157  for (int i = 0; i < EP_TABLE; i++) {
158  ctx->ep2_ptr[i] = &(ctx->ep2_pre[i]);
159  }
160 #endif
161 
162 #if ALLOC == STATIC || ALLOC == DYNAMIC || ALLOC == STACK
163  ctx->ep2_g.x[0] = ctx->ep2_gx[0];
164  ctx->ep2_g.x[1] = ctx->ep2_gx[1];
165  ctx->ep2_g.y[0] = ctx->ep2_gy[0];
166  ctx->ep2_g.y[1] = ctx->ep2_gy[1];
167  ctx->ep2_g.z[0] = ctx->ep2_gz[0];
168  ctx->ep2_g.z[1] = ctx->ep2_gz[1];
169 #endif
170 
171 #ifdef EP_PRECO
172 #if ALLOC == STATIC || ALLOC == DYNAMIC
173  for (int i = 0; i < EP_TABLE; i++) {
174  fp2_new(ctx->ep2_pre[i].x);
175  fp2_new(ctx->ep2_pre[i].y);
176  fp2_new(ctx->ep2_pre[i].z);
177  }
178 #elif ALLOC == STACK
179  for (int i = 0; i < EP_TABLE; i++) {
180  ctx->ep2_pre[i].x[0] = ctx->_ep2_pre[3 * i][0];
181  ctx->ep2_pre[i].x[1] = ctx->_ep2_pre[3 * i][1];
182  ctx->ep2_pre[i].y[0] = ctx->_ep2_pre[3 * i + 1][0];
183  ctx->ep2_pre[i].y[1] = ctx->_ep2_pre[3 * i + 1][1];
184  ctx->ep2_pre[i].z[0] = ctx->_ep2_pre[3 * i + 2][0];
185  ctx->ep2_pre[i].z[1] = ctx->_ep2_pre[3 * i + 2][1];
186  }
187 #endif
188 #endif
189  ep2_set_infty(&(ctx->ep2_g));
190  bn_init(&(ctx->ep2_r), FP_DIGS);
191 }

246 void ep2_curve_set(int twist) {
247  char str[2 * FP_BYTES + 1];
248  ctx_t *ctx = core_get();
249  ep2_t g;
250  fp2_t a;
251  fp2_t b;
252  bn_t r;
253 
254  ep2_null(g);
255  fp2_null(a);
256  fp2_null(b);
257  bn_null(r);
258 
259  TRY {
260  ep2_new(g);
261  fp2_new(a);
262  fp2_new(b);
263  bn_new(r);
264 
265  switch (ep_param_get()) {
266 #if FP_PRIME == 158
267  case BN_P158:
268  ASSIGN(BN_P158);
269  break;
270 #elif FP_PRIME == 254
271  case BN_P254:
272  ASSIGN(BN_P254);
273  break;
274 #elif FP_PRIME == 256
275  case BN_P256:
276  ASSIGN(BN_P256);
277  break;
278 #elif FP_PRIME == 638
279  case BN_P638:
280  ASSIGN(BN_P638);
281  break;
282  case B12_P638:
283  ASSIGN(B12_P638);
284  break;
285 #endif
286  default:
287  (void)str;
288  THROW(ERR_NO_VALID);
289  break;
290  }
291 
292  fp2_zero(g->z);
293  fp_set_dig(g->z[0], 1);
294  g->norm = 1;
295 
296  ctx->ep2_is_twist = twist;
297 
298  ep2_copy(&(ctx->ep2_g), g);
299  fp_copy(ctx->ep2_a[0], a[0]);
300  fp_copy(ctx->ep2_a[1], a[1]);
301  fp_copy(ctx->ep2_b[0], b[0]);
302  fp_copy(ctx->ep2_b[1], b[1]);
303  bn_copy(&(ctx->ep2_r), r);
304 
305  /* I don't have a better place for this. */
306  fp_prime_calc();
307 
308 #if defined(EP_PRECO)
309  ep2_mul_pre(ep2_curve_get_tab(), &(ctx->ep2_g));
310 #endif
311  }
312  CATCH_ANY {
313  THROW(ERR_CAUGHT);
314  }
315  FINALLY {
316  ep2_free(g);
317  fp2_free(a);
318  fp2_free(b);
319  bn_free(r);
320  }
