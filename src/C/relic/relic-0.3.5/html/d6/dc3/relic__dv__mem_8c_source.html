51 void dv_new_dynam(dv_t *a, int digits) {
52  if (digits > DV_DIGS) {
53  THROW(ERR_NO_PRECI);
54  }
55 #if ALIGN == 1
56  *a = malloc(digits * (DIGIT / 8));
57 #elif OPSYS == WINDOWS
58  *a = _aligned_malloc(digits * (DIGIT / 8), ALIGN);
59 #else
60  int r = posix_memalign((void **)a, ALIGN, digits * (DIGIT / 8));
61  if (r == ENOMEM) {
62  THROW(ERR_NO_MEMORY);
63  }
64  if (r == EINVAL) {
65  THROW(ERR_NO_CONFIG);
66  }
67 #endif
68 
69  if (*a == NULL) {
70  THROW(ERR_NO_MEMORY);
71  }
72 }
