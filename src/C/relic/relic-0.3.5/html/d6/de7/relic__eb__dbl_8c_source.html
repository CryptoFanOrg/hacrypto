51 static void eb_dbl_basic_ordin(eb_t r, const eb_t p) {
52  fb_t t0, t1, t2;
53 
54  fb_null(t0);
55  fb_null(t1);
56  fb_null(t2);
57 
58  TRY {
59  fb_new(t0);
60  fb_new(t1);
61  fb_new(t2);
62 
63  /* t0 = 1/x1. */
64  fb_inv(t0, p->x);
65  /* t0 = y1/x1. */
66  fb_mul(t0, t0, p->y);
67  /* t0 = lambda = x1 + y1/x1. */
68  fb_add(t0, t0, p->x);
69  /* t1 = lambda^2. */
70  fb_sqr(t1, t0);
71  /* t2 = lambda^2 + lambda. */
72  fb_add(t2, t1, t0);
73 
74  /* t2 = lambda^2 + lambda + a2. */
75  switch (eb_curve_opt_a()) {
76  case OPT_ZERO:
77  break;
78  case OPT_ONE:
79  fb_add_dig(t2, t2, (dig_t)1);
80  break;
81  case OPT_DIGIT:
82  fb_add_dig(t2, t2, eb_curve_get_a()[0]);
83  break;
84  default:
85  fb_add(t2, t2, eb_curve_get_a());
86  break;
87  }
88 
89  /* t1 = x1 + x3. */
90  fb_add(t1, t2, p->x);
91 
92  /* t1 = lambda * (x1 + x3). */
93  fb_mul(t1, t0, t1);
94 
95  fb_copy(r->x, t2);
96  /* y3 = lambda * (x1 + x3) + x3 + y1. */
97  fb_add(t1, t1, r->x);
98  fb_add(r->y, t1, p->y);
99 
100  fb_copy(r->z, p->z);
101 
102  r->norm = 1;
103  }
104  CATCH_ANY {
105  THROW(ERR_CAUGHT);
106  }
107  FINALLY {
108  fb_free(t0);
109  fb_free(t1);
110  fb_free(t2);
111  }
112 }

217 static void eb_dbl_projc_ordin(eb_t r, const eb_t p) {
218  fb_t t0, t1;
219 
220  fb_null(t0);
221  fb_null(t1);
222 
223  TRY {
224  fb_new(t0);
225  fb_new(t1);
226 
227  /* t0 = B = x1^2. */
228  fb_sqr(t0, p->x);
229  /* C = B + y1. */
230  fb_add(r->y, t0, p->y);
231 
232  if (!p->norm) {
233  /* A = x1 * z1. */
234  fb_mul(t1, p->x, p->z);
235  /* z3 = A^2. */
236  fb_sqr(r->z, t1);
237  } else {
238  /* if z1 = 1, A = x1. */
239  fb_copy(t1, p->x);
240  /* if z1 = 1, z3 = x1^2. */
241  fb_copy(r->z, t0);
242  }
243 
244  /* t1 = D = A * C. */
245  fb_mul(t1, t1, r->y);
246 
247  /* C^2 + D. */
248  fb_sqr(r->y, r->y);
249  fb_add(r->x, t1, r->y);
250 
251  /* C^2 + D + a2 * z3. */
252  switch (eb_curve_opt_a()) {
253  case OPT_ZERO:
254  break;
255  case OPT_ONE:
256  fb_add(r->x, r->z, r->x);
257  break;
258  case OPT_DIGIT:
259  fb_mul_dig(r->y, r->z, eb_curve_get_a()[0]);
260  fb_add(r->x, r->y, r->x);
261  break;
262  default:
263  fb_mul(r->y, r->z, eb_curve_get_a());
264  fb_add(r->x, r->y, r->x);
265  break;
266  }
267 
268  /* t1 = (D + z3). */
269  fb_add(t1, t1, r->z);
270  /* t0 = B^2. */
271  fb_sqr(t0, t0);
272  /* t0 = B^2 * z3. */
273  fb_mul(t0, t0, r->z);
274  /* y3 = (D + z3) * r3 + B^2 * z3. */
275  fb_mul(r->y, t1, r->x);
276  fb_add(r->y, r->y, t0);
277 
278  r->norm = 0;
279  }
280  CATCH_ANY {
281  THROW(ERR_CAUGHT);
282  }
283  FINALLY {
284  fb_free(t0);
285  fb_free(t1);
286  }
287 }

298 static void eb_dbl_projc_super(eb_t r, const eb_t p) {
299  /* x3 = x1^4 */
300  fb_sqr(r->x, p->x);
301  fb_sqr(r->x, r->x);
302  /* y3 = y1^4. */
303  fb_sqr(r->y, p->y);
304  fb_sqr(r->y, r->y);
305 
306  if (!p->norm) {
307  /* z3 = z1^4. */
308  fb_sqr(r->z, p->z);
309  fb_sqr(r->z, r->z);
310  /* y3 = x1^4 + y1^4. */
311  fb_add(r->y, r->x, r->y);
312  /* x3 = x1^4 + z1^4. */
313  fb_add(r->x, r->x, r->z);
314  } else {
315  /* y3 = x1^4 + y1^4. */
316  fb_add(r->y, r->x, r->y);
317  /* x3 = x1^4 + 1. */
318  fb_add_dig(r->x, r->x, 1);
319  /* r is still in affine coordinates. */
320  fb_copy(r->z, p->z);
321  }
322 
323  r->norm = p->norm;
324 }
