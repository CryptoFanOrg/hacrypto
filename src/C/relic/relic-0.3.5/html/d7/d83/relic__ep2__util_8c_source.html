72 void ep2_rand(ep2_t p) {
73  bn_t n, k;
74  ep2_t gen;
75 
76  bn_null(k);
77  bn_null(n);
78  ep2_null(gen);
79 
80  TRY {
81  bn_new(k);
82  bn_new(n);
83  ep2_new(gen);
84 
85  ep2_curve_get_ord(n);
86 
87  bn_rand(k, BN_POS, bn_bits(n));
88  bn_mod(k, k, n);
89 
90  ep2_curve_get_gen(gen);
91  ep2_mul(p, gen, k);
92  }
93  CATCH_ANY {

95  }
96  FINALLY {
97  bn_free(k);
98  bn_free(n);
99  ep2_free(gen);

103 void ep2_rhs(fp2_t rhs, ep2_t p) {
104  fp2_t t0;
105  fp2_t t1;
106 
107  fp2_null(t0);
108  fp2_null(t1);
109 
110  TRY {
111  fp2_new(t0);
112  fp2_new(t1);
113 
114  /* t0 = x1^2. */
115  fp2_sqr(t0, p->x);
116  /* t1 = x1^3. */
117  fp2_mul(t1, t0, p->x);
118 
119  ep2_curve_get_a(t0);
120  fp2_mul(t0, p->x, t0);
121  fp2_add(t1, t1, t0);
122 
123  ep2_curve_get_b(t0);
124  fp2_add(t1, t1, t0);
125 
126  fp2_copy(rhs, t1);
127 
128  } CATCH_ANY {
129  THROW(ERR_CAUGHT);
130  } FINALLY {
131  fp2_free(t0);
132  fp2_free(t1);
133  }
134 }

136 void ep2_tab(ep2_t * t, ep2_t p, int w) {
137  if (w > 2) {
138  ep2_dbl(t[0], p);
139 #if defined(EP_MIXED)
140  ep2_norm(t[0], t[0]);
141 #endif
142  ep2_add(t[1], t[0], p);
143  for (int i = 2; i < (1 << (w - 2)); i++) {
144  ep2_add(t[i], t[i - 1], t[0]);
145  }
146 #if defined(EP_MIXED)
147  for (int i = 1; i < (1 << (EP_WIDTH - 2)); i++) {
148  ep2_norm(t[i], t[i]);
149  }
150 #endif
151  }
152  ep2_copy(t[0], p);
153 }

155 int ep2_is_valid(ep2_t p) {
156  ep2_t t;
157  int r = 0;
158 
159  ep2_null(t);
160 
161  TRY {
162  ep2_new(t);
163 
164  ep2_norm(t, p);
165 
166  ep2_rhs(t->x, t);
167  fp2_sqr(t->y, t->y);
168 
169  r = (fp2_cmp(t->x, t->y) == CMP_EQ) || ep2_is_infty(p);
170  } CATCH_ANY {
171  THROW(ERR_CAUGHT);
172  } FINALLY {
173  ep2_free(t);
174  }
175  return r;
176 }
