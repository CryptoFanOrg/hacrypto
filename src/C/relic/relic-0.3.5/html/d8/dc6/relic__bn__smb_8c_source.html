38 void bn_smb_leg(bn_t c, const bn_t a, const bn_t b) {
39  bn_t t;
40 
41  bn_null(t);
42 
43  if (bn_cmp(a, b) == CMP_EQ) {
44  bn_zero(c);
45  return;
46  }
47 
48  TRY {
49  bn_new(t);
50 
51  if (bn_sign(b) == BN_NEG) {
52  THROW(ERR_NO_VALID);
53  }
54 
55  /* t = (b - 1)/2. */
56  bn_sub_dig(t, b, 1);
57  bn_rsh(t, t, 1);
58 #if BN_MOD == MONTY
59  bn_mxp(c, a, t, b);
60 #else
61  bn_mxp(c, a, t, b);
62 #endif
63  bn_sub_dig(t, b, 1);
64  if (bn_cmp(c, t) == CMP_EQ) {
65  bn_set_dig(c, 1);
66  bn_neg(c, c);
67  }
68  }
69  CATCH_ANY {
70  THROW(ERR_CAUGHT);
71  }
72  FINALLY {
73  bn_free(t);
74  }
75 }

77 void bn_smb_jac(bn_t c, const bn_t a, const bn_t b) {
78  bn_t t0, t1, r;
79  int t, h;
80 
81  bn_null(t0);
82  bn_null(t1);
83  bn_null(r);
84 
85  TRY {
86  bn_new(t0);
87  bn_new(t1);
88  bn_new(r);
89  t = 1;
90 
91  /* Argument b must be odd. */
92  if (bn_is_even(b) || bn_sign(b) == BN_NEG) {
93  THROW(ERR_NO_VALID);
94  }
95 
96  if (bn_sign(a) == BN_NEG) {
97  bn_add(t0, a, b);
98  } else {
99  bn_copy(t0, a);
100  }
101  bn_copy(t1, b);
102 
103  while (1) {
104  /* t0 = a mod b. */
105  bn_mod(t0, t0, t1);
106  /* If a = 0 then if n = 1 return t else return 0. */
107  if (bn_is_zero(t0)) {
108  if (bn_cmp_dig(t1, 1) == CMP_EQ) {
109  bn_set_dig(c, 1);
110  if (t == -1) {
111  bn_neg(c, c);
112  }
113  break;
114  } else {
115  bn_zero(c);
116  break;
117  }
118  }
119  /* Write t0 as 2^h * t0. */
120  h = 0;
121  while (bn_is_even(t0)) {
122  h++;
123  bn_rsh(t0, t0, 1);
124  }
125  /* If h != 0 (mod 2) and n != +-1 (mod 8) then t = -t. */
126  bn_mod_2b(r, t1, 3);
127  if ((h % 2 != 0) && (bn_cmp_dig(r, 1) != CMP_EQ) &&
128  (bn_cmp_dig(r, 7) != CMP_EQ)) {
129  t = -t;
130  }
131  /* If t0 != 1 (mod 4) and n != 1 (mod 4) then t = -t. */
132  bn_mod_2b(r, t0, 2);
133  if (bn_cmp_dig(r, 1) != CMP_EQ) {
134  bn_mod_2b(r, t1, 2);
135  if (bn_cmp_dig(r, 1) != CMP_EQ) {
136  t = -t;
137  }
138  }
139  bn_copy(r, t0);
140  bn_copy(t0, t1);
141  bn_copy(t1, r);
142  }
143  }
144  CATCH_ANY {
145  THROW(ERR_CAUGHT);
146  }
147  FINALLY {
148  bn_free(t0);
149  bn_free(t1);
150  bn_free(r);
151  }
152 }
