44 static void detect_opt(int *opt, fp_t a) {
45  fp_t t;
46 
47  fp_null(t);
48 
49  TRY {
50  fp_new(t);
51  fp_prime_conv_dig(t, 3);
52  fp_neg(t, t);
53 
54  if (fp_cmp(a, t) == CMP_EQ) {
55  *opt = OPT_MINUS3;
56  } else {
57  if (fp_is_zero(a)) {
58  *opt = OPT_ZERO;
59  } else {
60  fp_set_dig(t, 1);
61  if (fp_cmp_dig(a, 1) == CMP_EQ) {
62  *opt = OPT_ONE;
63  } else {
64  if (fp_cmp_dig(a, 2) == CMP_EQ) {
65  *opt = OPT_TWO;
66  } else {
67  if (fp_bits(a) <= FP_DIGIT) {
68  *opt = OPT_DIGIT;
69  } else {
70  *opt = OPT_NONE;
71  }
72  }
73  }
74  }
75  }
76  }
77  CATCH_ANY {
78  THROW(ERR_CAUGHT);
79  }
80  FINALLY {
81  fp_free(t);
82  }
83 }

269 void ep_curve_set_kbltz(const fp_t b, const ep_t g, const bn_t r, const bn_t h,
270  const fp_t beta, const bn_t l) {
271  int bits = bn_bits(r);
272  ctx_t *ctx = core_get();
273  ctx->ep_is_kbltz = 1;
274  ctx->ep_is_super = 0;
275 
276  fp_zero(ctx->ep_a);
277  fp_copy(ctx->ep_b, b);
278 
279  detect_opt(&(ctx->ep_opt_a), ctx->ep_a);
280  detect_opt(&(ctx->ep_opt_b), ctx->ep_b);
281 
282 #if EP_MUL == LWNAF || EP_FIX == COMBS || EP_FIX == LWNAF || EP_SIM == INTER || !defined(STRIP)
283  fp_copy(ctx->beta, beta);
284  bn_gcd_ext_mid(&(ctx->ep_v1[1]), &(ctx->ep_v1[2]), &(ctx->ep_v2[1]),
285  &(ctx->ep_v2[2]), l, r);
286  /* l = v1[1] * v2[2] - v1[2] * v2[1], r = l / 2. */
287  bn_mul(&(ctx->ep_v1[0]), &(ctx->ep_v1[1]), &(ctx->ep_v2[2]));
288  bn_mul(&(ctx->ep_v2[0]), &(ctx->ep_v1[2]), &(ctx->ep_v2[1]));
289  bn_sub(&(ctx->ep_r), &(ctx->ep_v1[0]), &(ctx->ep_v2[0]));
290  bn_hlv(&(ctx->ep_r), &(ctx->ep_r));
291  /* v1[0] = round(v2[2] * 2^|n| / l). */
292  bn_lsh(&(ctx->ep_v1[0]), &(ctx->ep_v2[2]), bits + 1);
293  if (bn_sign(&(ctx->ep_v1[0])) == BN_POS) {
294  bn_add(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), &(ctx->ep_r));
295  } else {
296  bn_sub(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), &(ctx->ep_r));
297  }
298  bn_dbl(&(ctx->ep_r), &(ctx->ep_r));
299  bn_div(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), &(ctx->ep_r));
300  /* v2[0] = round(v1[2] * 2^|n| / l). */
301  bn_lsh(&(ctx->ep_v2[0]), &(ctx->ep_v1[2]), bits + 1);
302  if (bn_sign(&(ctx->ep_v2[0])) == BN_POS) {
303  bn_add(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), &(ctx->ep_r));
304  } else {
305  bn_sub(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), &(ctx->ep_r));
306  }
307  bn_div(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), &(ctx->ep_r));
308  bn_neg(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]));
309 #endif
310 
311  ep_norm(&(ctx->ep_g), g);
312  bn_copy(&(ctx->ep_r), r);
313  bn_copy(&(ctx->ep_h), h);
314 
315 #if defined(EP_PRECO)
316  ep_mul_pre((ep_t *)ep_curve_get_tab(), &(ctx->ep_g));
317 #endif
318 }
