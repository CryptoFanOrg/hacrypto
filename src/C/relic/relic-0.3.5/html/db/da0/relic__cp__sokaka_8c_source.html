47 void cp_sokaka_gen(bn_t master) {
48  bn_t n;
49 
50  bn_null(n);
51 
52  TRY {
53  bn_new(n);
54 
55  g1_get_ord(n);
56 
57  do {
58  bn_rand(master, BN_POS, bn_bits(n));
59  bn_mod(master, master, n);
60  } while (bn_is_zero(master));
61  }
62  CATCH_ANY {
63  THROW(ERR_CAUGHT);
64  }
65  FINALLY {
66  bn_free(n);
67  }
68 }
