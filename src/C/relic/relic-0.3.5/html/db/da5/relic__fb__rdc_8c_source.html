50 void fb_rdc_basic(fb_t c, dv_t a) {
51  int j, k;
52  dig_t *tmpa;
53  dv_t r;
54 
55  dv_null(r);
56 
57  TRY {
58  dv_new(r);
59 
60  tmpa = a + FB_DIGS;
61 
62  /* First reduce the high part. */
63  for (int i = fb_bits(tmpa) - 1; i >= 0; i--) {
64  if (fb_test_bit(tmpa, i)) {
65  SPLIT(k, j, i - FB_BITS, FB_DIG_LOG);
66  if (k <= 0) {
67  fb_addd_low(tmpa + j, tmpa + j, fb_poly_get(), FB_DIGS);
68  } else {
69  r[FB_DIGS] = fb_lshb_low(r, fb_poly_get(), k);
70  fb_addd_low(tmpa + j, tmpa + j, r, FB_DIGS + 1);
71  }
72  }
73  }
74  for (int i = fb_bits(a) - 1; i >= FB_BITS; i--) {
75  if (fb_test_bit(a, i)) {
76  SPLIT(k, j, i - FB_BITS, FB_DIG_LOG);
77  if (k == 0) {
78  fb_addd_low(a + j, a + j, fb_poly_get(), FB_DIGS);
79  } else {
80  r[FB_DIGS] = fb_lshb_low(r, fb_poly_get(), k);
81  fb_addd_low(a + j, a + j, r, FB_DIGS + 1);
82  }
83  }
84  }
85 
86  fb_copy(c, a);
87  }
88  CATCH_ANY {
89  THROW(ERR_CAUGHT);
90  }
91  FINALLY {
92  fb_free(r);
93  }
94 }
