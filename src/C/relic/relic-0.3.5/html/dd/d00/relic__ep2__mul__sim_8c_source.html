41 static void ep2_mul_sim_ordin(ep2_t r, ep2_t p, bn_t k, ep2_t q, bn_t l, int gen) {
42  int len, l0, l1, i, n0, n1, w;
43  signed char naf0[FP_BITS + 1], naf1[FP_BITS + 1], *t0, *t1;
44  ep2_t table0[1 << (EP_WIDTH - 2)];
45  ep2_t table1[1 << (EP_WIDTH - 2)];
46  ep2_t *t = NULL;
47 
48  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
49  ep2_null(table0[i]);
50  ep2_null(table1[i]);
51  }
52 
53  if (gen) {
54 #if defined(EP_PRECO)
55  t = ep2_curve_get_tab();
56 #endif
57  } else {
58  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
59  ep2_new(table0[i]);
60  }
61  ep2_tab(table0, p, EP_WIDTH);
62  t = table0;
63  }
64 
65  /* Prepare the precomputation table. */
66  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
67  ep2_new(table1[i]);
68  }
69  /* Compute the precomputation table. */
70  ep2_tab(table1, q, EP_WIDTH);
71 
72  /* Compute the w-TNAF representation of k. */
73  if (gen) {
74  w = EP_DEPTH;
75  } else {
76  w = EP_WIDTH;
77  }
78 
79  l0 = l1 = FP_BITS + 1;
80  bn_rec_naf(naf0, &l0, k, w);
81  bn_rec_naf(naf1, &l1, l, EP_WIDTH);
82 
83  len = MAX(l0, l1);
84  t0 = naf0 + len - 1;
85  t1 = naf1 + len - 1;
86  for (i = l0; i < len; i++)
87  naf0[i] = 0;
88  for (i = l1; i < len; i++)
89  naf1[i] = 0;
90 
91  ep2_set_infty(r);
92  for (i = len - 1; i >= 0; i--, t0--, t1--) {
93  ep2_dbl(r, r);
94 
95  n0 = *t0;
96  n1 = *t1;
97  if (n0 > 0) {
98  ep2_add(r, r, t[n0 / 2]);
99  }
100  if (n0 < 0) {
101  ep2_sub(r, r, t[-n0 / 2]);
102  }
103  if (n1 > 0) {
104  ep2_add(r, r, table1[n1 / 2]);
105  }
106  if (n1 < 0) {
107  ep2_sub(r, r, table1[-n1 / 2]);
108  }
109  }
110  /* Convert r to affine coordinates. */
111  ep2_norm(r, r);
112 
113  /* Free the precomputation table. */
114  for (i = 0; i < 1 << (EP_WIDTH - 2); i++) {
115  ep2_free(table0[i]);
116  ep2_free(table1[i]);
117  }
118 }

201  bn_rec_win(w0, &l0, k, w);
202  bn_rec_win(w1, &l1, l, w);
203 
204  for (int i = l0; i < l1; i++) {
205  w0[i] = 0;
206  }
207  for (int i = l1; i < l0; i++) {
208  w1[i] = 0;
209  }

213  for (int j = 0; j < w; j++) {
214  ep2_dbl(r, r);
215  }

217  }
218  ep2_norm(r, r);
219  } CATCH_ANY {

221  }
222  FINALLY {

229  ep2_free(t[i]);

255 void ep2_mul_sim_joint(ep2_t r, ep2_t p, bn_t k, ep2_t q, bn_t l) {
256  ep2_t t[5];
257  int u_i, len, offset;
258  signed char jsf[2 * (FP_BITS + 1)];
259  int i;
260 
261  ep2_null(t[0]);
262  ep2_null(t[1]);

264  ep2_null(t[3]);
265  ep2_null(t[4]);

271

273  ep2_copy(t[1], q);
274  ep2_copy(t[2], p);
275  ep2_add(t[3], p, q);
276  ep2_sub(t[4], p, q);
277 
278  len = 2 * (FP_BITS + 1);
279  bn_rec_jsf(jsf, &len, k, l);
280 
281  ep2_set_infty(r);
282 
283  i = bn_bits(k);
284  offset = MAX(i, bn_bits(l)) + 1;
285  for (i = len - 1; i >= 0; i--) {
286  ep2_dbl(r, r);
287  if (jsf[i] != 0 && jsf[i] == -jsf[i + offset]) {
288  u_i = jsf[i] * 2 + jsf[i + offset];
289  if (u_i < 0) {
290  ep2_sub(r, r, t[4]);
291  } else {
292  ep2_add(r, r, t[4]);
293  }
294  } else {
295  u_i = jsf[i] * 2 + jsf[i + offset];
296  if (u_i < 0) {
297  ep2_sub(r, r, t[-u_i]);
298  } else {
299  ep2_add(r, r, t[u_i]);
300  }
301  }
302  }
303  ep2_norm(r, r);
304  }
305  CATCH_ANY {
306  THROW(ERR_CAUGHT);
307  }
308  FINALLY {
309  for (i = 0; i < 5; i++) {
310  ep2_free(t[i]);
311  }
312  }
313 }
