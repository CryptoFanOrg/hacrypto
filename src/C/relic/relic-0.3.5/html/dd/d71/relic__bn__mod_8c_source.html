40 void bn_mod_2b(bn_t c, const bn_t a, int b) {
41  int i, first, d;
42 
43  if (b <= 0) {
44  bn_zero(c);
45  return;
46  }
47 
48  if (b >= (int)(a->used * BN_DIGIT)) {
49  bn_copy(c, a);
50  return;
51  }
52 
53  bn_copy(c, a);
54 
55  SPLIT(b, d, b, BN_DIG_LOG);
56 
57  first = (d) + (b == 0 ? 0 : 1);
58  for (i = first; i < c->used; i++)
59  c->dp[i] = 0;
60 
61  c->dp[d] &= MASK(b);
62 
63  bn_trim(c);
64 }

81 void bn_mod_barrt(bn_t c, const bn_t a, const bn_t m, const bn_t u) {
82  unsigned long mu;
83  bn_t q, t;
84 
85  bn_null(q);
86  bn_null(t);
87 
88  if (bn_cmp(a, m) == CMP_LT) {
89  bn_copy(c, a);
90  return;
91  }
92  TRY {
93  bn_new(q);
94  bn_new(t);
95  bn_zero(t);
96 
97  mu = m->used;
98 
99  bn_rsh(q, a, (mu - 1) * BN_DIGIT);
100 
101  if (mu > ((dig_t)1) << (BN_DIGIT - 1)) {
102  bn_mul(t, q, u);
103  } else {
104  if (q->used > u->used) {
105  bn_muld_low(t->dp, q->dp, q->used, u->dp, u->used,
106  mu, q->used + u->used);
107  } else {
108  bn_muld_low(t->dp, u->dp, u->used, q->dp, q->used,
109  mu - (u->used - q->used), q->used + u->used);
110  }
111  t->used = q->used + u->used;
112  bn_trim(t);
113  }
114 
115  bn_rsh(q, t, (mu + 1) * BN_DIGIT);
116 
117  if (q->used > m->used) {
118  bn_muld_low(t->dp, q->dp, q->used, m->dp, m->used, 0, q->used + 1);
119  } else {
120  bn_muld_low(t->dp, m->dp, m->used, q->dp, q->used, 0, mu + 1);
121  }
122  t->used = mu + 1;
123  bn_trim(t);
124 
125  bn_mod_2b(q, t, BN_DIGIT * (mu + 1));
126  bn_mod_2b(t, a, BN_DIGIT * (mu + 1));
127  bn_sub(t, t, q);
128 
129  if (bn_sign(t) == BN_NEG) {
130  bn_set_dig(q, (dig_t)1);
131  bn_lsh(q, q, (mu + 1) * BN_DIGIT);
132  bn_add(t, t, q);
133  }
134 
135  while (bn_cmp(t, m) != CMP_LT) {
136  bn_sub(t, t, m);
137  }
138 
139  bn_copy(c, t);
140  }
141  CATCH_ANY {
142  THROW(ERR_CAUGHT);
143  }
144  FINALLY {
145  bn_free(q);
146  bn_free(t);
147 
148  }
149 }

155 void bn_mod_pre_monty(bn_t u, const bn_t m) {
156  dig_t x, b;
157  b = m->dp[0];
158 
159  if ((b & 0x01) == 0) {
160  THROW(ERR_NO_VALID);
161  }
162 
163  x = (((b + 2) & 4) << 1) + b; /* here x*a==1 mod 2**4 */
164  x *= 2 - b * x; /* here x*a==1 mod 2**8 */
165 #if WORD != 8
166  x *= 2 - b * x; /* here x*a==1 mod 2**16 */
167 #endif
168 #if WORD == 64 || WORD != 8 || WORD == 16
169  x *= 2 - b * x; /* here x*a==1 mod 2**32 */
170 #endif
171 #if WORD == 64
172  x *= 2 - b * x; /* here x*a==1 mod 2**64 */
173 #endif
174  /* u = -1/m0 (mod 2^BN_DIGIT) */
175  bn_set_dig(u, -x);
176 }

207 void bn_mod_monty_basic(bn_t c, const bn_t a, const bn_t m, const bn_t u) {
208  int digits, i;
209  dig_t r, carry, u0, *tmp;
210  bn_t t;
211 
212  bn_null(t);
213  digits = 2 * m->used + 1;
214  TRY {
215  bn_new_size(t, digits);
216  bn_zero(t);
217  bn_copy(t, a);
218  t->used = digits;
219 
220  u0 = u->dp[0];
221  tmp = t->dp;
222 
223  for (i = 0; i < m->used; i++, tmp++) {
224  r = (dig_t)(*tmp * u0);
225  carry = bn_mula_low(tmp, m->dp, r, m->used);
226  bn_add1_low(tmp + m->used, tmp + m->used, carry, m->used - i + 1);
227  }
228  bn_rsh(t, t, m->used * BN_DIGIT);
229  bn_trim(t);
230 
231  if (bn_cmp_abs(t, m) != CMP_LT) {
232  bn_sub(t, t, m);
233  }
234 
235  bn_copy(c, t);
236  }
237  CATCH_ANY {
238  THROW(ERR_CAUGHT);
239  }
240  FINALLY {
241  bn_free(t);
242  }
243 }
